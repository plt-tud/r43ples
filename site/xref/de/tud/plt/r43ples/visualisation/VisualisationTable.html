25  public class VisualisationTable {
26  	
27  	private static final int TIMELINEVIEW_WIDTH = 70;
28  
29  	/**
30  	 * Vertical distance between commits
31  	 */
32  	protected final static int LINE_HEIGHT = 22;
33  	
34  	protected final static int PADDING = 10;
35  	
36  	private TimeLineView timeView;
37  
38  	private List<Commit> commits;
39  
40  	private CommitGraphView graphView;
41  
42  	private MessagesTableView msgView;
43  	
44  
45  	public VisualisationTable(StructuredTree revisionTree) {
46  		commits = revisionTree.getCommits();
47  		Collections.reverse(commits);
48  		
49  		timeView = new TimeLineView(commits);
50  		graphView = new CommitGraphView(commits);
51  		msgView = new MessagesTableView(revisionTree);
52  	}
53  	
54  	
55  	private String buildSVG() throws SVGGraphics2DIOException {
56  		
57  		// generate svg document
58  		String svgNS = SVGDOMImplementation.SVG_NAMESPACE_URI;
59  		DOMImplementation impl = SVGDOMImplementation.getDOMImplementation();
60  		Document doc = impl.createDocument(svgNS, "svg", null);
61  
62  		SVGGraphics2D g = new SVGGraphics2D(doc);
63  		
64  
65  		// translate to first text-baseline
66  		g.translate(0, LINE_HEIGHT);		
67  		timeView.draw(g);
68  
69  		g.translate(TIMELINEVIEW_WIDTH, 0);
70  		graphView.drawGraph(g);
71  
72  		g.translate(graphView.getDimension().getWidth() + PADDING, 0);
73  		msgView.draw(g);
74  		
75  		g.translate(-graphView.getDimension().getWidth() - PADDING - TIMELINEVIEW_WIDTH, 0);
76  
77  		// calculate overall width
78  		int totalWidth = (int) (TIMELINEVIEW_WIDTH + graphView.getDimension().getWidth() + PADDING + msgView.getDimension().getWidth());
79  		
80  		
81  		// draw header line
82  		int offset = g.getFontMetrics(msgView.TextFont).getDescent();
83  		g.drawLine(0, offset, totalWidth, offset);
84  
85  		int totalHeight = LINE_HEIGHT * (commits.size()+1) + offset;
86  		g.setSVGCanvasSize(new Dimension(totalWidth, totalHeight));
87  
88  		Writer writer = new StringWriter();
89  		g.stream(writer);
90  
91  		return writer.toString();
92  	}
93  
94  	public static String getHtmlOutput(String graphName) {
95  		MustacheFactory mf = new DefaultMustacheFactory();
96  		Mustache mustache = mf.compile("templates/graphvisualisation_table.mustache");
97  		StringWriter sw = new StringWriter();
98  
99  		// get graph tree
100 		RevisionGraph graph = new RevisionGraph(graphName);
101 		String revisionGraphUri = graph.getRevisionGraphUri();
102 		StructuredTree graphTree = StructuredTree.getTreeOfGraph(revisionGraphUri);
103 		VisualisationTable visu = new VisualisationTable(graphTree);
104 		
105 		Map<String, Object> scope = new HashMap<String, Object>();
106 		scope.put("graphName", graphName);
107 		try {
108 			scope.put("svg_content", visu.buildSVG());
109 		} catch (SVGGraphics2DIOException e) {
110 			scope.put("svg_content", "Error while creating SVG");
111 		}
112 
113 		mustache.execute(sw, scope);
114 		return sw.toString();
115 	}
