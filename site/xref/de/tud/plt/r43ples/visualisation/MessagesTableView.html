18  public class MessagesTableView {
19  
20  	/**
21  	 * Font of the Text
22  	 */
23  	protected final Font TextFont = new Font("SansSerif", Font.PLAIN, 16);
24  
25  	private List<Commit> commits;
26  	private Map<Commit, List<Branch>> branch_index;
27  	private Map<Commit, List<Tag>> tag_index;
28  	private Dimension dimension = null;
29  
30  	public MessagesTableView(StructuredTree tree) {
31  		commits = tree.getCommits();
32  
33  		// generate branches index
34  		branch_index = new HashMap<Commit, List<Branch>>();
35  		for (Branch b : tree.getBranches()) {
36  			if (branch_index.containsKey(b.getReference()))
37  				branch_index.get(b.getReference()).add(b);
38  			else {
39  				List<Branch> list = new LinkedList<Branch>();
40  				list.add(b);
41  				branch_index.put(b.getReference(), list);
42  			}
43  		}
44  
45  		// generate tags index
46  		tag_index = new HashMap<Commit, List<Tag>>();
47  		for (Tag t : tree.getTags()) {
48  			if (tag_index.containsKey(t.getReference()))
49  				tag_index.get(t.getReference()).add(t);
50  			else {
51  				List<Tag> list = new LinkedList<Tag>();
52  				list.add(t);
53  				tag_index.put(t.getReference(), list);
54  			}
55  		}
56  	}
57  
58  	public void draw(Graphics2D g) {
59  
60  		int y = 0;
61  		int pos = 0;
62  		FontMetrics fm = g.getFontMetrics(TextFont);
63  		Font tmp = g.getFont();
64  		g.setColor(Color.BLACK);
65  		g.setFont(TextFont);
66  		Font branchTagFont = new Font("Monospaced", Font.PLAIN, TextFont.getSize() - 2);
67  		FontMetrics branchTagMetrics = g.getFontMetrics(branchTagFont);
68  		
69  		// Number
70  		// header line
71  		g.drawString("Number", pos, y);
72  		int numberWidth = fm.stringWidth("Number");
73  		y = VisualisationTable.LINE_HEIGHT;
74  
75  		for (Commit c : commits) {
76  			
77  			String rev = c.getNextRevision();
78  			g.drawString(rev, pos, y);
79  
80  			// calculate offset of next column
81  			numberWidth = Math.max(numberWidth, fm.stringWidth(rev));
82  
83  			y += VisualisationTable.LINE_HEIGHT;
84  		}
85  		
86  		
87  		pos += numberWidth + VisualisationTable.PADDING;
88  		
89  		// Commit message
90  		// header line
91  		y=0;
92  		g.drawString("Commit Message", pos, y);
93  		int commitWidth = fm.stringWidth("Commit Message");
94  		y += VisualisationTable.LINE_HEIGHT;
95  
96  		for (Commit c : commits) {
97  			String message = c.getMessage();
98  
99  			int branchesWidth = 0;
100 			g.setFont(branchTagFont);
101 
102 			// draw branches
103 			if (branch_index.containsKey(c)) {
104 				for (Branch b : branch_index.get(c)) {
105 					String name = b.getName();
106 
107 					int width = branchTagMetrics.stringWidth(name);
108 					int height = branchTagMetrics.getAscent() + branchTagMetrics.getDescent();
109 					g.translate(0, y);
110 					g.setColor(new Color(0.5f, 0.1f, 0.1f));
111 					g.fillRoundRect(pos + branchesWidth, -branchTagMetrics.getAscent(), width + 8, height, 5, 5);
112 					g.setColor(Color.WHITE);
113 					g.drawString(name, pos + branchesWidth + 2, 0);
114 					g.translate(0, -y);
115 					branchesWidth += width + 12;
116 				}
117 			}
118 
119 			// draw tags
120 			if (tag_index.containsKey(c)) {
121 				for (Tag t : tag_index.get(c)) {
122 					String name = t.getName();
123 
124 					int width = branchTagMetrics.stringWidth(name);
125 					int height = branchTagMetrics.getAscent() + branchTagMetrics.getDescent();
126 					g.translate(0, y);
127 					g.setColor(new Color(0.1f, 0.1f, 0.5f));
128 					g.fillRoundRect(pos + branchesWidth, -branchTagMetrics.getAscent(), width + 8, height, 5, 5);
129 					g.setColor(Color.WHITE);
130 					g.drawString(name, pos + branchesWidth + 2, 0);
131 					g.translate(0, -y);
132 					branchesWidth += width + 12;
133 				}
134 			}
135 
136 			g.setColor(Color.BLACK);
137 			g.setFont(TextFont);
138 
139 			// infos of commit
140 			g.drawString(message, pos + branchesWidth, y);
141 
142 			// calculate offset of next column
143 			commitWidth = Math.max(commitWidth, branchesWidth + fm.stringWidth(message));
144 
145 			y += VisualisationTable.LINE_HEIGHT;
146 		}
147 		
148 		pos += commitWidth + VisualisationTable.PADDING;
149 
150 		
151 		// Author
152 		y = 0;
153 
154 		// header line
155 		g.drawString("Author", pos, y);
156 		int authorWidth = fm.stringWidth("Author");
157 		y += VisualisationTable.LINE_HEIGHT;
158 
159 		for (Commit c : commits) {
160 			String author = c.getAuthor();
161 			g.drawString(author, pos, y);
162 
163 			// calculate offset of next column
164 			authorWidth = Math.max(authorWidth, fm.stringWidth(author));
165 
166 			y += VisualisationTable.LINE_HEIGHT;
167 		}
168 		pos += authorWidth;
169 		dimension = new Dimension(pos, y);
170 
171 		g.setFont(tmp);
172 	}
173 
174 	public Dimension getDimension() {
175 		return dimension;
176 	}

178 }
