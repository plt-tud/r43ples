25  /**
26   * Coevolve all dependent graphs based upon aggregated atomic changes. Uses the coevolution rules provided which are associated with the HLC aggregation.
27   * Example query: USER "Bob" MESSAGE "test coevolution" COEVO GRAPH <http://test.com/r43ples-dataset-hlc-aggregation> REVISION "1" TO REVISION "2"
28   * Currently the implementation only works if the end revision is the direct succeeding revision of the start revision.
29   * Dependent graphs currently only identified on their master branch. The result of the coevolution is currently always committed to the master branch.
30   *
31   * @author Stephan Hensel
32   */
33  public class CoEvolutionDraft {
34  
35      /** The logger. **/
36      private Logger logger = LogManager.getLogger(SelectConstructAskQuery.class);
37  
38      /** The pattern modifier. **/
39      private final int patternModifier = Pattern.DOTALL + Pattern.MULTILINE + Pattern.CASE_INSENSITIVE;
40      /** The merge query pattern. **/
41      private final Pattern patternAggQuery = Pattern.compile(
42              "COEVO\\s*GRAPH\\s*<(?<graph>[^>]*?)>\\s*REVISION\\s*\"(?<startRevisionIdentifier>[^\"]*?)\"\\s*TO\\s*REVISION\\s*\"(?<endRevisionIdentifier>[^\"]*?)\"\\s*",
43              patternModifier);
44  
45      /** The corresponding R43ples request. **/
46      private R43plesRequest request;
47      /** The associated user name of the commit. **/
48      private String user;
49      /** The message of the commit. **/
50      private String message;
51      /** The start revision identifier. **/
52      private String startRevisionIdentifier;
53      /** The end revision identifier. **/
54      private String endRevisionIdentifier;
55      /** The change set between start and end revision. **/
56      private ChangeSet changeSetStartToEnd;
57      /** The graph name **/
58      private String graphName;
59      /** The revision graph. **/
60      private RevisionGraph revisionGraph;

88      /**
89       * Triggers the coevolution process and writes meta information into revision graph and the coevolution graph.
90       *
91       * @return the evolution object containing all performed coevolutions
92       * @throws InternalErrorException
93       */
94      protected Evolution coevolveAll() throws InternalErrorException {
95  
96          // Stores all meta information of the evolution, has to be integrated into a SPARQL UPDATE query if all data is collected
97          StringBuilder metaInformationTurtle = new StringBuilder();
98          String evolutionURI = uriCalculator.getRandomURI(Config.evolution_graph);
99  
100         metaInformationTurtle.append(String.format(
101                 "<%1$s> a rmo:Evolution. %n" +
102                 "<%1$s> rmo:startRevision <%2$s>. %n" +
103                 "<%1$s> rmo:endRevision <%3$s>. %n" +
104                 "<%1$s> rmo:usedSourceRevisionGraph <%4$s>. %n",
105                 evolutionURI, revisionGraph.getRevision(startRevisionIdentifier).getRevisionURI(),
106                 revisionGraph.getRevision(endRevisionIdentifier).getRevisionURI(), revisionGraph.getGraphName()));
107 
108         // Get associated semantic changes with specified change set
109         LinkedList<SemanticChange> semanticChanges = changeSetStartToEnd.getSemanticChangesList();
110 
111         // Check if semantic changes are not generated
112         if ((semanticChanges == null) || (semanticChanges.isEmpty())) {
113             // Aggregate
114             AggregationDraft aggregationDraft = new AggregationDraft(revisionGraph, changeSetStartToEnd);
115             semanticChanges = aggregationDraft.aggregate();
116             if ((semanticChanges == null) || (semanticChanges.isEmpty())) {
117                 throw new InternalErrorException("There are no semantic changes between revision " + startRevisionIdentifier + " and revision " + endRevisionIdentifier + "as basis for the coevolution.");
118             }
119         }
120 
121         // Coevolution rule list
122         LinkedList<CoEvoRule> coevolutionRules = new LinkedList<>();
123 
124         // Iterate through all available semantic changes and store results within the coevolution rule list
125         for (SemanticChange semanticChange : semanticChanges) {
126             // Get the rule of the semantic change and check if a coevolution part is specified
127             CoEvoRule coEvoRule = new CoEvoRule(semanticChange);
128             coevolutionRules.add(coEvoRule);
129             metaInformationTurtle.append(String.format("<%s> rmo:associatedSemanticChange <%s>. %n", evolutionURI, semanticChange.getSemanticChangeURI()));
130         }
131 
132         // Get all graphs within the repository
133         HashMap<String, RevisionGraph> revisedGraphs = new RevisionControl().getRevisedGraphs();
134         // Remove the graph which should be coevolved and the coevolution graph itself
135         revisedGraphs.remove(this.revisionGraph.getGraphName());
136         revisedGraphs.remove(Config.evolution_graph);
137 
138         // Iterate through the revised graphs and search for dependencies (Check the master branch of each revised graph if there is a dependency)
139         for (String graphName : revisedGraphs.keySet()) {
140             RevisionGraph revisionGraph = revisedGraphs.get(graphName);
141 
142             // URI of the coevolution of the current revision graph
143             String coevolutionURI = uriCalculator.getRandomURI(Config.evolution_graph);
144             metaInformationTurtle.append(String.format(
145                     "<%1$s> rmo:performedCoEvolution <%2$s>. %n" +
146                     "<%2$s> a rmo:CoEvolution. %n" +
147                     "<%2$s> rmo:usedTargetRevisionGraph <%3$s>. %n" +
148                     "<%2$s> rmo:usedTargetBranch <%4$s>. %n",
149                     evolutionURI, coevolutionURI, revisionGraph.getGraphName(), revisionGraph.getBranchUri("master")));
150 
151             // Create temporary named graphs for add and delete
152             String tempAddSetURI = uriCalculator.getRandomNamedGraphURI(graphName);
153             String tempDeleteSetURI = uriCalculator.getRandomNamedGraphURI(graphName);
154 
155             tripleStoreInterface.executeCreateGraph(tempAddSetURI);
156             tripleStoreInterface.executeCreateGraph(tempDeleteSetURI);
157 
158             // Iterate through all available semantic changes and apply the corresponding coevolution match part of the rule
159             for (CoEvoRule coEvoRule : coevolutionRules) {
160 
161                 String matchQuery = coEvoRule.getDependencyMatchingQuery().replaceAll("<http://NAMEDGRAPH#master>", "<" + graphName + ">");
162                 ResultSet resultSetMatchings = tripleStoreInterface.executeSelectQuery(matchQuery);
163 
164                 // If the result set is not equal to null a coevolution can be executed
165                 // URI of the coevolution of the current revision graph
166                 String appliedCoevolutionURI = uriCalculator.getRandomURI(Config.evolution_graph);
167                 metaInformationTurtle.append(String.format(
168                         "<%1$s> aero:appliedCoEvolutionRule <%2$s>. %n" +
169                         "<%2$s> a aero:AppliedCoEvolutionRule. %n" +
170                         "<%2$s> aero:usedRule <%3$s>. %n" +
171                         "<%2$s> aero:usedSemanticChange <%4$s>. %n",
172                         coevolutionURI, appliedCoevolutionURI, coEvoRule.getSemanticChange().getUsedRuleURI(), coEvoRule.getSemanticChange().getSemanticChangeURI()));
173 
174                 // Maybe there are multiple matches within one graph
175                 while (resultSetMatchings.hasNext()) {
176                     QuerySolution qsMatching = resultSetMatchings.next();
177 
178                     String sparqlVariableGroupURI = uriCalculator.getRandomURI(Config.evolution_graph);
179                     metaInformationTurtle.append(String.format(
180                             "<%1$s> aero:hasVariableGroup <%2$s>. %n" +
181                             "<%2$s> a aero:SPARQLVariableGroup. %n",
182                             appliedCoevolutionURI, sparqlVariableGroupURI));
183 
184                     // Create a list of SPARQL variables and the results regarding the current revision graph
185                     LinkedList<SparqlVariable> sparqlVariablesListCurrentRevisionGraph = new LinkedList<>();
186 
187                     String addSetInsertQuery = coEvoRule.getAddSetInsertQuery();
188                     addSetInsertQuery = addSetInsertQuery.replaceAll("<http://NAMEDGRAPH#master>", "<" + graphName + ">");
189                     addSetInsertQuery = addSetInsertQuery.replaceAll("<http://NAMEDGRAPH#ADDSET-NEW>", "<" + tempAddSetURI + ">");
190                     String deleteSetInsertQuery = coEvoRule.getDeleteSetInsertQuery();
191                     deleteSetInsertQuery = deleteSetInsertQuery.replaceAll("<http://NAMEDGRAPH#master>", "<" + graphName + ">");
192                     deleteSetInsertQuery = deleteSetInsertQuery.replaceAll("<http://NAMEDGRAPH#DELETESET-NEW>", "<" + tempDeleteSetURI + ">");
193 
194                     for (SparqlVariable genericSparqlVariable : coEvoRule.getSparqlVariablesList()) {
195                         String value;
196                         String valueMetaInfo;
197                         boolean isResource;
198                         try {
199                             value = qsMatching.getResource("?" + genericSparqlVariable.getVariableName()).toString();
200                             valueMetaInfo = "<" + value + ">";
201                             isResource = true;
202                         } catch (Exception e) {
203                             value = qsMatching.getLiteral("?" + genericSparqlVariable.getVariableName()).toString();
204                             valueMetaInfo = "\"" + value + "\"";
205                             isResource = false;
206                         }
207                         SparqlVariable specificSparqlVariable = new SparqlVariable(revisionGraph, null, genericSparqlVariable.getVariableName(), genericSparqlVariable.getSpinResourceURI(), value, isResource);
208                         sparqlVariablesListCurrentRevisionGraph.add(specificSparqlVariable);
209 
210                         addSetInsertQuery = addSetInsertQuery.replaceAll("\\?" + genericSparqlVariable.getVariableName(), valueMetaInfo);
211                         deleteSetInsertQuery = deleteSetInsertQuery.replaceAll("\\?" + genericSparqlVariable.getVariableName(), valueMetaInfo);
212 
213                         tripleStoreInterface.executeUpdateQuery(addSetInsertQuery);
214                         tripleStoreInterface.executeUpdateQuery(deleteSetInsertQuery);
215 
216                         String sparqlVariableURI = uriCalculator.getRandomURI(Config.evolution_graph);
217                         metaInformationTurtle.append(String.format(
218                                 "<%1$s> aero:hasVariables <%2$s>. %n" +
219                                 "<%2$s> a aero:SPARQLVariable. %n" +
220                                 "<%2$s> sp:varName \"%3$s\". %n" +
221                                 "<%2$s> aero:value %4$s. %n" +
222                                 "<%2$s> aero:spinResource <%5$s>. %n",
223                                 sparqlVariableGroupURI, sparqlVariableURI, genericSparqlVariable.getVariableName(), valueMetaInfo, genericSparqlVariable.getSpinResourceURI()));
224 
225                     }
226                 }
227 
228             }
229 
230             String addSetN3 = Helper.getContentOfNamedGraphAsN3(tempAddSetURI);
231             String deleteSetN3 = Helper.getContentOfNamedGraphAsN3(tempDeleteSetURI);
232 
233             // Create a new update commit with the specified add and delete sets from temporary graphs for the current revision graph
234             Branch branch = new Branch(revisionGraph, "master", true);
235 
236             UpdateCommitDraft updateCommitDraft = new UpdateCommitDraft(graphName, addSetN3, deleteSetN3, user, message, branch);
237             UpdateCommit updateCommit = updateCommitDraft.createInTripleStore().get(0);
238 
239             tripleStoreInterface.executeUpdateQuery("DROP SILENT GRAPH <" + tempAddSetURI + ">");
240             tripleStoreInterface.executeUpdateQuery("DROP SILENT GRAPH <" + tempDeleteSetURI + ">");
241 
242             // Create the generated revision
243             metaInformationTurtle.append(String.format(
244                     "<%1$s> rmo:generated <%2$s>. %n",
245                     coevolutionURI, updateCommit.getGeneratedRevision().getRevisionURI()));
246         }
247 
248         // Create a new commit within the coevolution revision graph
249         String metaInformationN3 = JenaModelManagement.convertTurtleToNTriple(Config.turtle_prefixes + metaInformationTurtle.toString());
250         R43plesCoreSingleton.getInstance().createUpdateCommit(Config.evolution_graph, metaInformationN3, null, user, message, "master");
251 
252         return new Evolution(evolutionURI);
253     }

255     /**
256      * Extracts the request information and stores it to local variables.
257      *
258      * @throws InternalErrorException
259      */
260     private void extractRequestInformation() throws InternalErrorException {
261 
262         // Reuse a CommitDraft to extract the user and the commit message
263         CommitDraft commitDraft = new CommitDraft(this.request);
264         this.user = commitDraft.getUser();
265         this.message = commitDraft.getMessage();
266 
267         Matcher m = patternAggQuery.matcher(this.request.query_sparql);
268 
269         boolean foundEntry = false;
270 
271         while (m.find()) {
272             foundEntry = true;
273 
274             graphName = m.group("graph");
275             revisionGraph = new RevisionGraph(graphName);
276 
277             startRevisionIdentifier = m.group("startRevisionIdentifier");
278             endRevisionIdentifier = m.group("endRevisionIdentifier");
279 
280             logger.debug("graph: " + graphName);
281             logger.debug("startRevisionIdentifier: " + startRevisionIdentifier);
282             logger.debug("endRevisionIdentifier: " + endRevisionIdentifier);
283         }
284         if (!foundEntry) {
285             throw new QueryErrorException("Error in query: " + this.request.query_sparql);
286         }
287 
288         ArrayList<ChangeSet> changeSets = revisionGraph.getRevision(endRevisionIdentifier).getChangeSets();
289         for (ChangeSet changeSet : changeSets) {
290             if (changeSet.getPriorRevision().getRevisionIdentifier().equals(startRevisionIdentifier)) {
291                 changeSetStartToEnd = changeSet;
292             }
293         }
294 
295     }
