16  /**
17   * Collection of information for creating a new tag commit.
18   *
19   * @author Stephan Hensel
20   */
21  public class TagCommitDraft extends ReferenceCommitDraft {
22  
23      /** The logger. **/
24      private Logger logger = LogManager.getLogger(TagCommitDraft.class);
25  
26      /** The pattern modifier. **/
27      private final int patternModifier = Pattern.DOTALL + Pattern.MULTILINE + Pattern.CASE_INSENSITIVE;
28  
29  
30      /**
31       * The constructor.
32       * Creates a tag commit draft by using the corresponding meta information.
33       *
34       * @param revisionGraph the revision graph
35       * @param referenceIdentifier the reference identifier
36       * @param baseRevision the revision identifier (the corresponding revision will be the current base for the reference)
37       * @param user the user
38       * @param message the message
39       * @throws InternalErrorException
40       */
41      protected TagCommitDraft(RevisionGraph revisionGraph, String referenceIdentifier, Revision baseRevision, String user, String message) throws InternalErrorException {
42          super(revisionGraph, referenceIdentifier, baseRevision, user, message, true);
43      }
44  
45      /**
46       * Creates the commit draft as a new commit in the triple store.
47       *
48       * @return the created tag commit
49       */
50      protected TagCommit createInTripleStore() throws InternalErrorException {
51          String commitURI = getUriCalculator().getNewTagCommitURI(getRevisionGraph(), getReferenceIdentifier());
52  
53          // Check tag existence
54          if (getRevisionGraph().hasReference(getReferenceIdentifier())) {
55              // Reference name is already in use
56              logger.error("The reference name '" + getReferenceIdentifier() + "' is for the graph '" + getRevisionGraph().getGraphName()
57                      + "' already in use.");
58              throw new IdentifierAlreadyExistsException("The reference name '" + getReferenceIdentifier()
59                      + "' is for the graph '" + getRevisionGraph().getGraphName() + "' already in use.");
60          }
61  
62          TagDraft tagDraft = new TagDraft(getUriCalculator(), getRevisionGraph(), getBaseRevision(), getReferenceIdentifier());
63          Tag generatedTag = tagDraft.createInTripleStore();
64  
65          addMetaInformation(generatedTag.getReferenceURI(), commitURI);
66  
67  
68  
69          return new TagCommit(getRevisionGraph(), commitURI, getUser(), getTimeStamp(), getMessage(), getBaseRevision(), generatedTag);
70      }
71  
72      /**
73       * Adds the necessary meta data.
74       *
75       * @param referenceURI the reference URI
76       * @param commitURI the commit URI
77       * @throws InternalErrorException
78       */
79      private void addMetaInformation(String referenceURI, String commitURI) {
80          logger.info("Create new tag '" + getReferenceIdentifier() + "' for graph " + getRevisionGraph().getGraphName());
81  
82          // General variables
83          String personUri = Helper.getUserURI(getUser());
84  
85          // Create a new commit (activity)
86          String queryContent = String.format(""
87                          + "<%s> a rmo:TagCommit, rmo:ReferenceCommit, rmo:Commit ; "
88                          + "	rmo:wasAssociatedWith <%s> ;"
89                          + "	rmo:generated <%s> ;"
90                          + " rmo:used <%s> ;"
91                          + "	rmo:commitMessage \"%s\" ;"
92                          + "	rmo:timeStamp \"%s\" .%n",
93                  commitURI, personUri, referenceURI, getBaseRevision().getRevisionURI(), getMessage(), getTimeStamp());
94  
95          // Execute queries
96          String query = Config.prefixes
97                  + String.format("INSERT DATA { GRAPH <%s> { %s } } ;", getRevisionGraph().getRevisionGraphUri(), queryContent);
98          getTripleStoreInterface().executeUpdateQuery(query);
99  
100     }
