265     /**
266      * Drop graph query. This query will delete the whole graph and all corresponding revision information.
267      *
268      * @param query the query
269      * @throws QueryErrorException
270     */
271     @Override
272     public void sparqlDropGraph(final String query) throws QueryErrorException {
273         /* The pattern modifier. */
274         int patternModifier = Pattern.DOTALL + Pattern.MULTILINE + Pattern.CASE_INSENSITIVE;
275         final Pattern patternDropGraph = Pattern.compile("DROP\\s*(?<silent>SILENT)?\\s*GRAPH\\s*<(?<graph>[^>]*)>",
276                 patternModifier);
277         Matcher m = patternDropGraph.matcher(query);
278         boolean found = false;
279         while (m.find()) {
280             found = true;
281             String graphName = m.group("graph");
282             RevisionGraph graph = new RevisionGraph(graphName);
283             graph.purgeRevisionInformation();
284         }
285         if (!found) {
286             throw new QueryErrorException("Query contain errors:\n" + query);
287         }
288     }

290     /**
291      * Get the response of a SPARQL query (SELECT, CONSTRUCT, ASK).
292      *
293      * @param request the request
294      * @param query_rewriting option if query rewriting should be enabled (true => enabled)
295      * @return the query response
296      * @throws InternalErrorException
297      */
298     @Override
299     public String getSparqlSelectConstructAskResponse(final R43plesRequest request, final boolean query_rewriting) throws InternalErrorException {
300         String result;
301         if (query_rewriting) {
302             String query_rewritten = SparqlRewriter.rewriteQuery(request.query_sparql);
303             result = TripleStoreInterfaceSingleton.get()
304                     .executeSelectConstructAskQuery(Config.getUserDefinedSparqlPrefixes() + query_rewritten, request.format);
305         } else {
306             SelectConstructAskQuery selectConstructAskQuery = new SelectConstructAskQuery(request);
307             result = selectConstructAskQuery.performQuery();
308         }
309         return result;
310     }
