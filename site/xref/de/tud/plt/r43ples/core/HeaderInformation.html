21      /** Creates an RDF description for the revision tree of the graphs specified in the given SPARQL query
22       * @param query SPARQL query
23       * @return RDF string containing information for graphs specified in query
24       */
25      public String getResponseHeaderFromQuery(String query) {
26          final Pattern patternGraph = Pattern.compile(
27                  "(GRAPH|FROM|INTO)\\s*<(?<graph>[^>]*)>\\s*",
28                  Pattern.CASE_INSENSITIVE);
29  
30          StringBuilder graphNames = new StringBuilder();
31          Matcher m = patternGraph.matcher(query);
32          m.find();
33          while (!m.hitEnd()) {
34              String graphName = m.group("graph");
35              graphNames.append("<"+graphName+">");
36              m.find();
37              if (!m.hitEnd())
38                  graphNames.append(", ");
39          }
40          String names = graphNames.toString();
41          String result = getResponseHeader(names);
42          logger.debug(String.format("Header information for %s: %s", query, result));
43          return result;
44  
45      }
46  
47      protected String getResponseHeader(String graphList) {
48          String queryConstruct = String.format(
49                      "PREFIX rmo:	<http://eatld.et.tu-dresden.de/rmo#> "
50                          + "PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#> %n"
51                          + "CONSTRUCT {"
52                          + " ?ref a ?type;"
53                          + "		rdfs:label ?label;"
54                          + "		rmo:references ?rev."
55                          + " ?rev rmo:revisionNumber ?number . %n"
56                          + "} %n"
57                          + "WHERE {"
58                          + " GRAPH <%s> {"
59                          + "   ?graph a rmo:RevisionGraph; rmo:hasRevisionGraph ?revisionGraph."
60                          + "   FILTER (?graph IN (%s))"
61                          + " }"
62                          + " GRAPH ?revisionGraph { "
63                          + " ?ref a ?type;"
64                              + "		rmo:referenceIdentifier ?label;%n"
65                          + "		rmo:references ?rev."
66                              + " ?rev rmo:revisionIdentifier ?number . %n"
67                          + "FILTER (?type IN (rmo:Tag, rmo:Master, rmo:Branch)) %n"
68                          + "} }", Config.revision_graph, graphList);
69          String header = TripleStoreInterfaceSingleton.get().executeConstructQuery(queryConstruct, FileUtils.langTurtle);
70          return header;
71      }
72  
73      protected void checkUpToDate(final String clientRevisionInformation, final String sparqlQuery) throws OutdatedException {
74  
75          Model clientModel = JenaModelManagement.readTurtleStringToJenaModel(clientRevisionInformation);
76  
77          String recentRevisionInformation = this.getResponseHeaderFromQuery(sparqlQuery);
78  
79          Model serverModel = JenaModelManagement.readTurtleStringToJenaModel(recentRevisionInformation);
80  
81          if (!clientModel.isIsomorphicWith(serverModel)) {
82               throw new OutdatedException(clientModel, serverModel);
83          }
84      }
