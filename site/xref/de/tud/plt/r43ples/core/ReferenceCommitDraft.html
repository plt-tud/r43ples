15  /**
16   * Collection of information for creating a new reference commit.
17   *
18   * @author Stephan Hensel
19   */
20  public class ReferenceCommitDraft extends CommitDraft {
21  
22      /** The logger. **/
23      private Logger logger = LogManager.getLogger(ReferenceCommitDraft.class);
24  
25      /** The pattern modifier. **/
26      private final int patternModifier = Pattern.DOTALL + Pattern.MULTILINE + Pattern.CASE_INSENSITIVE;
27  
28      /** The revision graph. **/
29      private RevisionGraph revisionGraph;
30      /** The reference identifier. **/
31      private String referenceIdentifier;
32      /** The base revision. **/
33      private Revision baseRevision;
34      /** States if the created reference is a branch or a tag. (branch => true; tag => false) **/
35      private boolean isBranch;
36  
37  
38      /**
39       * The constructor.
40       *
41       * @param request the request received by R43ples
42       * @throws InternalErrorException
43       */
44      public ReferenceCommitDraft(R43plesRequest request) throws InternalErrorException {
45          super(request);
46          this.extractRequestInformation();
47      }
48  
49      /**
50       * The constructor.
51       * Creates a reference commit draft by using the corresponding meta information.
52       *
53       * @param revisionGraph the revision graph
54       * @param referenceIdentifier the reference identifier
55       * @param baseRevision the revision identifier (the corresponding revision will be the current base for the reference)
56       * @param user the user
57       * @param message the message
58       * @param isBranch states if the created reference is a branch or a tag. (branch => true; tag => false)
59       * @throws InternalErrorException
60       */
61      protected ReferenceCommitDraft(RevisionGraph revisionGraph, String referenceIdentifier, Revision baseRevision, String user, String message, boolean isBranch) throws InternalErrorException {
62          super(null);
63          this.setUser(user);
64          this.setMessage(message);
65          this.revisionGraph = revisionGraph;
66          this.referenceIdentifier = referenceIdentifier;
67          this.baseRevision = baseRevision;
68          this.isBranch = isBranch;
69      }
70  
71      /**
72       * Extracts the request information and stores it to local variables.
73       *
74       * @throws InternalErrorException
75       */
76      private void extractRequestInformation() throws InternalErrorException {
77          final Pattern patternBranchOrTagQuery = Pattern.compile("(?<action>TAG|BRANCH)\\s*GRAPH\\s*<(?<graph>[^>]*)>\\s*REVISION\\s*\"(?<revision>[^\"]*)\"\\s*TO\\s*\"(?<name>[^\"]*)\"",
78                  patternModifier);
79          Matcher m = patternBranchOrTagQuery.matcher(getRequest().query_sparql);
80  
81          boolean foundEntry = false;
82  
83          while (m.find()) {
84              foundEntry = true;
85              String action = m.group("action");
86              this.revisionGraph = new RevisionGraph(m.group("graph"));
87              this.baseRevision = new Revision(this.revisionGraph, m.group("revision"), true);
88              this.referenceIdentifier = m.group("name");
89              if (action.equals("TAG")) {
90                  this.isBranch = false;
91              } else if (action.equals("BRANCH")) {
92                  this.isBranch = true;
93              } else {
94                  throw new QueryErrorException("Error in query: " + getRequest().query_sparql);
95              }
96          }
97          if (!foundEntry) {
98              throw new QueryErrorException("Error in query: " + getRequest().query_sparql);
99          }
100     }
101 
102     /**
103      * Creates the commit draft as a new commit in the triple store.
104      *
105      * @return the created reference commit
106      */
107     protected ReferenceCommit createInTripleStore() throws InternalErrorException {
108         if (isBranch) {
109             BranchCommitDraft branchCommitDraft = new BranchCommitDraft(revisionGraph, referenceIdentifier, baseRevision, getUser(), getMessage());
110             return branchCommitDraft.createInTripleStore();
111         } else {
112             TagCommitDraft tagCommitDraft = new TagCommitDraft(revisionGraph, referenceIdentifier, baseRevision, getUser(), getMessage());
113             return tagCommitDraft.createInTripleStore();
114         }
115     }

123         return revisionGraph;
124     }

131     public String getReferenceIdentifier() {
132         return referenceIdentifier;
133     }
134 
135     /**
136      * Get the base revision.
137      *
138      * @return the base revision
139      */
140     public Revision getBaseRevision() {
141         return baseRevision;
142     }
