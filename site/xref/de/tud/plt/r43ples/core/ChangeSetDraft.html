135     /**
136      * Creates the change set draft with all meta data and named graphs in the triplestore.
137      * Creates add and delete sets, strips the resulting sets if necessary and adds meta information.
138      *
139      * @return the created change set
140      */
141     protected ChangeSet createInTripleStore() {
142         logger.debug("Create new change set for graph " + revisionGraph.getGraphName() + ".");
143 
144         createAddAndDeleteSetsInTripleStore();
145         if (!isStripped) {
146             stripSets();
147         }
148         addMetaInformation();
149 
150         return new ChangeSet(revisionGraph, priorRevision, addSetURI, deleteSetURI, changeSetURI);
151     }
152 
153     /**
154      * Creates the add and delete sets in the triplestore.
155      */
156     private void createAddAndDeleteSetsInTripleStore() {
157         if (addSet!=null && !addSet.isEmpty()) {
158             // Create new named graph for add set
159             logger.debug("Create new graph with name " + addSetURI + ".");
160             tripleStoreInterface.executeUpdateQuery(String.format("CREATE SILENT GRAPH <%s>%n", addSetURI));
161             Helper.executeINSERT(addSetURI, addSet);
162         }
163 
164         if (deleteSet!=null && !deleteSet.isEmpty()) {
165             // Create new named graph for delete set
166             logger.debug("Create new graph with name " + deleteSetURI + ".");
167             tripleStoreInterface.executeUpdateQuery(String.format("CREATE SILENT GRAPH <%s>%n", deleteSetURI));
168             Helper.executeINSERT(deleteSetURI, deleteSet);
169         }
170 
171         if (isSpecifiedByRewrittenQuery) {
172             logger.debug("Create new graph with name " + addSetURI + " without content.");
173             tripleStoreInterface.executeUpdateQuery(String.format("CREATE SILENT GRAPH <%s>%n", addSetURI));
174             logger.debug("Create new graph with name " + deleteSetURI + " without content.");
175             tripleStoreInterface.executeUpdateQuery(String.format("CREATE SILENT GRAPH <%s>%n", deleteSetURI));
176         }
177     }
178 
179     /**
180      * Strips the add and delete sets. Removes duplicated data regarding the full graph.
181      */
182     private void stripSets() {
183         // remove doubled data
184         // (already existing triples in add set; not existing triples in delete set)
185         tripleStoreInterface.executeUpdateQuery(String.format(
186                 "DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }",
187                 addSetURI, referencedFullGraphURI));
188         tripleStoreInterface.executeUpdateQuery(String.format(
189                 "DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } MINUS { GRAPH <%s> { ?s ?p ?o. } } }",
190                 deleteSetURI, deleteSetURI, referencedFullGraphURI));
191     }

193     /**
194      * Adds meta information of the created revision to the revision graph.
195      *
196      * @throws InternalErrorException
197      */
198     private void addMetaInformation() {
199         // The prior revision could be null because of the initial commit
200         String queryContent;
201         if (priorRevision != null) {
202             // Create new revision
203             queryContent = String.format(
204                     "<%s> a rmo:ChangeSet;"
205                             + "	rmo:addSet <%s> ;"
206                             + "	rmo:deleteSet <%s> ;"
207                             + "	rmo:priorRevision <%s> ;"
208                             + " rmo:succeedingRevision <%s> .",
209                     changeSetURI, addSetURI, deleteSetURI, priorRevision.getRevisionURI(), newRevisionURI);
210         } else {
211             // Create new revision without prior revision
212             queryContent = String.format(
213                     "<%s> a rmo:ChangeSet;"
214                             + "	rmo:addSet <%s> ;"
215                             + "	rmo:deleteSet <%s> ;"
216                             + " rmo:succeedingRevision <%s> .",
217                     changeSetURI, addSetURI, deleteSetURI, newRevisionURI);
218         }
219 
220         String queryRevision = Config.prefixes + String.format("INSERT DATA { GRAPH <%s> {%s} }", revisionGraph.getRevisionGraphUri(), queryContent);
221 
222         tripleStoreInterface.executeUpdateQuery(queryRevision);
223     }
