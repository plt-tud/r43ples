5   package de.tud.plt.r43ples.core;
6   
7   import de.tud.plt.r43ples.exception.InternalErrorException;
8   import de.tud.plt.r43ples.existentobjects.ChangeSet;
9   import de.tud.plt.r43ples.existentobjects.Revision;
10  import de.tud.plt.r43ples.existentobjects.RevisionGraph;
11  import de.tud.plt.r43ples.optimization.ChangeSetPath;
12  import de.tud.plt.r43ples.optimization.PathCalculationFabric;
13  import de.tud.plt.r43ples.triplestoreInterface.TripleStoreInterfaceSingleton;
14  import org.apache.logging.log4j.LogManager;
15  import org.apache.logging.log4j.Logger;
16  
17  import java.util.LinkedList;
18  
19  public class FullGraph {
20  
21      private final String fullGraphUri;
22      private final Revision revision;
23      private final RevisionGraph revisionGraph;
24      private boolean created = false;
25  
26      /**
27       * The logger.
28       **/
29      private Logger logger = LogManager.getLogger(FullGraph.class);
30  
31  
32      /**
33       * Creates the whole revision from the add and delete sets of the
34       * predecessors. Saved in a new graph.
35       * Already existent graphs will be dropped and recreated.
36       *
37       * @param revisionGraph the revision graph
38       * @param revision      the revision
39       * @throws InternalErrorException
40       */
41      public FullGraph(RevisionGraph revisionGraph, Revision revision) throws InternalErrorException {
42  
43          this.revisionGraph = revisionGraph;
44          this.revision = revision;
45  
46          if (revision.getAssociatedBranch() != null) {
47              this.fullGraphUri = revision.getAssociatedBranch().getFullGraphURI();
48              this.logger.info("Full graph of revision " + revision.getRevisionIdentifier() + " of graph <" + revisionGraph.getGraphName()
49                      + "> already exists in <" + this.fullGraphUri + ">");
50          } else {
51              this.fullGraphUri = revision.getRevisionURI() + "-fullGraph";
52              this.createNewFullGraph();
53          }
54      }
55  
56      /**
57       * Creates the whole revision from the add and delete sets of the
58       * predecessors. Saved in a new graph with specified name.
59       * Already existent graphs will be dropped and recreated.
60       *
61       * @param revisionGraph the revision graph
62       * @param revision      the revision
63       * @param fullGraphUri  URI where full graph should be created
64       * @throws InternalErrorException
65       */
66      public FullGraph(RevisionGraph revisionGraph, Revision revision, String fullGraphUri) throws InternalErrorException {
67  
68          this.revisionGraph = revisionGraph;
69          this.revision = revision;
70          this.fullGraphUri = fullGraphUri;
71          this.createNewFullGraph();
72      }
73  
74  
75      protected void createNewFullGraph() throws InternalErrorException {
76  
77          this.logger.info("Rebuild whole content of revision " + revision.getRevisionIdentifier() + " of graph <" + revisionGraph.getGraphName()
78                  + "> into temporary graph <" + this.fullGraphUri + ">");
79  
80          // Create temporary graph
81          TripleStoreInterfaceSingleton.get().executeUpdateQuery("DROP SILENT GRAPH <" + fullGraphUri + ">");
82          TripleStoreInterfaceSingleton.get().executeUpdateQuery("CREATE GRAPH <" + fullGraphUri + ">");
83  
84          // Create path to revision
85          ChangeSetPath changeSetPath = PathCalculationFabric.getInstance(revisionGraph).getPathOfChangeSets(revision);
86  
87          // Copy branch to temporary graph
88          Revision currentRevision = changeSetPath.getTargetRevision();
89          String copyQuery = "COPY GRAPH <" + currentRevision.getAssociatedReference().getFullGraphURI() + "> TO GRAPH <" + fullGraphUri + ">";
90          TripleStoreInterfaceSingleton.get().executeUpdateQuery(copyQuery);
91  
92          // Apply changesets
93          LinkedList<ChangeSet> list = changeSetPath.getRevisionPath();
94          while (!list.isEmpty()) {
95              ChangeSet currentChangeSet = list.remove();
96  
97              // Add data to temporary graph
98              String graph_deleted = currentChangeSet.getDeleteSetURI();
99              this.applyDeleteSet(graph_deleted);
100 
101             // Remove data from temporary graph (no opposite of SPARQL ADD available)
102             String graph_added = currentChangeSet.getAddSetURI();
103             this.applyAddSet(graph_added);
104         }
105         this.created = true;
106     }
107 
108     /**
109      * Applies delete set stored in named graph graph_deleted
110      *
111      * @param graph_deleted Uri of named graph containing delete set
112      */
113     protected void applyDeleteSet(String graph_deleted) {
114         String deleteQuery = String.format("ADD SILENT GRAPH <%s> TO GRAPH <%s>", graph_deleted, this.fullGraphUri);
115         TripleStoreInterfaceSingleton.get().executeUpdateQuery(deleteQuery);
116     }
117 
118 
119     /**
120      * Applies add set stored in named graph graph_added
121      *
122      * @param graph_added Uri of named graph containing add set
123      */
124     protected void applyAddSet(String graph_added) {
125         String addQuery = String.format("" +
126                 "DELETE { GRAPH <%s> {?s ?p ?o.} } " +
127                 "WHERE  { GRAPH <%s> {?s ?p ?o.} }", this.fullGraphUri, graph_added);
128         TripleStoreInterfaceSingleton.get().executeUpdateQuery(addQuery);
129     }
130 
131 
132     public String getFullGraphUri() {
133         return fullGraphUri;
134     }
135 
136     public Revision getRevision() {
137         return revision;
138     }
139 
140     public RevisionGraph getRevisionGraph() {
141         return revisionGraph;
142     }
143 
144     public void purge() {
145         if (this.created) {
146             String deleteQuery = String.format("DROP SILENT GRAPH <%s>", this.fullGraphUri);
147             TripleStoreInterfaceSingleton.get().executeUpdateQuery(deleteQuery);
148         }
149     }
150 }
