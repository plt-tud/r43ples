15  /**
16   * Collection of information for creating a new branch commit.
17   *
18   * @author Stephan Hensel
19   */
20  public class BranchCommitDraft extends ReferenceCommitDraft {
21  
22      /** The logger. **/
23      private Logger logger = LogManager.getLogger(BranchCommitDraft.class);
24  
25      /** The pattern modifier. **/
26      private final int patternModifier = Pattern.DOTALL + Pattern.MULTILINE + Pattern.CASE_INSENSITIVE;
27  
28  
29      /**
30       * The constructor.
31       * Creates a branch commit draft by using the corresponding meta information.
32       *
33       * @param revisionGraph the revision graph
34       * @param referenceIdentifier the reference identifier
35       * @param baseRevision the revision identifier (the corresponding revision will be the current base for the reference)
36       * @param user the user
37       * @param message the message
38       * @throws InternalErrorException
39       */
40      protected BranchCommitDraft(RevisionGraph revisionGraph, String referenceIdentifier, Revision baseRevision, String user, String message) throws InternalErrorException {
41          super(revisionGraph, referenceIdentifier, baseRevision, user, message, true);
42      }
43  
44      /**
45       * Creates the commit draft as a new commit in the triple store.
46       *
47       * @return the created branch commit
48       */
49      protected BranchCommit createInTripleStore() throws InternalErrorException {
50          String commitURI = getUriCalculator().getNewBranchCommitURI(getRevisionGraph(), getReferenceIdentifier());
51  
52          BranchDraft branchDraft = new BranchDraft(getUriCalculator(), getRevisionGraph(), getBaseRevision(), getReferenceIdentifier());
53          Branch generatedBranch = branchDraft.createInTripleStore();
54  
55  
56          addMetaInformation(generatedBranch.getReferenceURI(), commitURI);
57  
58          return new BranchCommit(getRevisionGraph(), commitURI, getUser(), getTimeStamp(), getMessage(), getBaseRevision(), generatedBranch);
59      }
60  
61      /**
62       * Adds the necessary meta data.
63       *
64       * @param referenceURI the reference URI
65       * @param commitURI the commit URI
66       * @throws InternalErrorException
67       */
68      private void addMetaInformation(String referenceURI, String commitURI) throws InternalErrorException {
69          logger.info("Create new branch '" + getReferenceIdentifier() + "' for graph " + getRevisionGraph().getGraphName());
70  
71          // General variables
72          String personUri = Helper.getUserURI(getUser());
73  
74          // Create a new commit (activity)
75          String queryContent = String.format(""
76                          + "<%s> a rmo:BranchCommit, rmo:ReferenceCommit, rmo:Commit ; "
77                          + "	rmo:wasAssociatedWith <%s> ;"
78                          + "	rmo:generated <%s> ;"
79                          + " rmo:used <%s> ;"
80                          + "	rmo:commitMessage \"%s\" ;"
81                          + "	rmo:timeStamp \"%s\" .%n",
82                  commitURI, personUri, referenceURI, getBaseRevision().getRevisionURI(), getMessage(), getTimeStamp());
83  
84          // Execute queries
85          String query = Config.prefixes
86                  + String.format("INSERT DATA { GRAPH <%s> { %s } } ;", getRevisionGraph().getRevisionGraphUri(), queryContent);
87          getTripleStoreInterface().executeUpdateQuery(query);
88      }

90  }
