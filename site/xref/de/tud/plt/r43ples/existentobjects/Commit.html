3   import org.apache.jena.query.QuerySolution;
4   import org.apache.jena.query.ResultSet;
5   import de.tud.plt.r43ples.exception.InternalErrorException;
6   import de.tud.plt.r43ples.management.Config;
7   import de.tud.plt.r43ples.triplestoreInterface.TripleStoreInterfaceSingleton;
8   import org.apache.logging.log4j.LogManager;
9   import org.apache.logging.log4j.Logger;
10  
11  /**
12   * Provides information of an already existent commit.
13   *
14   * @author Stephan Hensel
15   */
16  public class Commit {
17  
18      /** The logger. **/
19      private Logger logger = LogManager.getLogger(Commit.class);
20  
21      /** The commit URI. */
22      private String commitURI;
23      /** The commit message. */
24      private String commitMessage;
25      /** The commit time stamp. */
26      private String commitTimeStamp;
27      /** The associated person name. */
28      private String commitAssociatedPersonName;
29  
30      /** The revision graph URI. */
31      private String revisionGraphURI;
32      /** The corresponding revision graph. */
33      private RevisionGraph revisionGraph;
34  
35  
36      /**
37       * The constructor.
38       *
39       * @param revisionGraph the revision graph
40       * @param commitURI the commit URI
41       * @throws InternalErrorException
42       */
43      public Commit(RevisionGraph revisionGraph, String commitURI) throws InternalErrorException {
44          this.revisionGraph = revisionGraph;
45          this.revisionGraphURI = this.revisionGraph.getRevisionGraphUri();
46          this.commitURI = commitURI;
47  
48          retrieveAdditionalInformation();
49      }
50  
51      /**
52       * The constructor.
53       *
54       * @param revisionGraph the revision graph
55       * @param commitURI the commit URI
56       * @param user the user
57       * @param timeStamp the time stamp
58       * @param message the message
59       * @throws InternalErrorException
60       */
61      public Commit(RevisionGraph revisionGraph, String commitURI, String user, String timeStamp, String message) throws InternalErrorException {
62          this.revisionGraph = revisionGraph;
63          this.revisionGraphURI = this.revisionGraph.getRevisionGraphUri();
64          this.commitURI = commitURI;
65          this.commitAssociatedPersonName = user;
66          this.commitTimeStamp = timeStamp;
67          this.commitMessage = message;
68      }
69  
70      /**
71       * Calculate additional information of the current commit and store this information to local variables.
72       *
73       * @throws InternalErrorException
74       */
75      private void retrieveAdditionalInformation() throws InternalErrorException {
76          logger.info("Get additional information of current commit URI " + commitURI + ".");
77          String query = Config.prefixes + String.format(""
78                  + "SELECT ?title ?time ?person "
79                  + "WHERE { GRAPH  <%s> {"
80                  + "	<%s> a rmo:Commit; "
81                  + "	 rmo:commitMessage ?title; "
82                  + "  rmo:timeStamp ?time; "
83                  + "  rmo:wasAssociatedWith ?person. "
84                  + "} }", revisionGraphURI, commitURI);
85          this.logger.debug(query);
86          ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
87          if (resultSet.hasNext()) {
88              QuerySolution qs = resultSet.next();
89              commitMessage = qs.getLiteral("?title").toString();
90              commitTimeStamp = qs.getLiteral("?timeStamp").toString();
91              commitAssociatedPersonName = qs.getResource("?person").toString();
92          } else {
93              throw new InternalErrorException("No additional information found for commit URI " + commitURI + ".");
94          }
95      }
96  
97      /**
98       * Get the commit URI.
99       *
100      * @return the commit URI
101      */
102     public String getCommitURI() {
103         return commitURI;
104     }
105 
106     /**
107      * Get the commit message.
108      *
109      * @return the commit message
110      */
111     public String getCommitMessage() {
112         return commitMessage;
113     }
114 
115     /**
116      * Get the commit time stamp.
117      *
118      * @return the commit time stamp
119      */
120     public String getCommitTimeStamp() {
121         return commitTimeStamp;
122     }
123 
124     /**
125      * Get the associated person name.
126      *
127      * @return the associated person name
128      */
129     public String getCommitAssociatedPersonName() {
130         return commitAssociatedPersonName;
131     }
132 
133     /**
134      * Get the revision graph.
135      *
136      * @return the revision graph
137      */
138     public RevisionGraph getRevisionGraph() {
139         return revisionGraph;
140     }
141 
142 }
