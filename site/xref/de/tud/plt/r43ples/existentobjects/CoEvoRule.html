67      /**
68       * Calculate additional information of the current commit and store this information to local variables.
69       *
70       * @throws InternalErrorException
71       */
72      private void retrieveAdditionalInformation() throws InternalErrorException {
73          logger.info("Get associated coevolution rule information of semantic change.");
74          // Get the rule of the semantic change and check if a coevolution part is specified
75          String ruleURI = semanticChange.getUsedRuleURI();
76          String queryCoEvoRule = String.format(
77                  Config.prefixes
78                          + "SELECT ?dependencyQuery ?addQuery ?delQuery %n"
79                          + "WHERE { GRAPH <%s> { %n"
80                          + "	<%s> a aero:CoEvoRule ; %n"
81                          + "   aero:dependencyMatchingQuery ?dependencyQuery; %n"
82                          + "   aero:addSetInsertQuery ?addQuery; %n"
83                          + "   aero:deleteSetInsertQuery ?delQuery. %n"
84                          + "} } %n", Config.rules_graph, ruleURI);
85  
86          ResultSet resultSetCoEvoRule = tripleStoreInterface.executeSelectQuery(queryCoEvoRule);
87          if (resultSetCoEvoRule.hasNext()) {
88              QuerySolution qsSpin = resultSetCoEvoRule.next();
89              this.spinDependencyMatchingQueryURI = qsSpin.getResource("?dependencyQuery").toString();
90              this.spinAddSetInsertQueryURI = qsSpin.getResource("?addQuery").toString();
91              this.spinDeleteSetInsertQueryURI = qsSpin.getResource("?delQuery").toString();
92          } else {
93              throw new InternalErrorException("No coevolution rules specified for semantic change.");
94          }
95  
96          // Get the SPARQL queries from SPIN URIs
97          String spinDependencyMatchingQueryN3 = Helper.getAllRelatedElementsToURI(Config.rules_graph, this.spinDependencyMatchingQueryURI);
98          String spinAddSetQueryN3 = Helper.getAllRelatedElementsToURI(Config.rules_graph, this.spinAddSetInsertQueryURI);
99          String spinDeleteSetQueryN3 = Helper.getAllRelatedElementsToURI(Config.rules_graph, this.spinDeleteSetInsertQueryURI);
100 
101         this.dependencyMatchingQuery = Helper.getSparqlSelectQueryFromSpin(spinDependencyMatchingQueryN3, this.spinDependencyMatchingQueryURI);
102         this.addSetInsertQuery = Helper.getSparqlUpdateQueryFromSpin(spinAddSetQueryN3, this.spinAddSetInsertQueryURI);
103         this.deleteSetInsertQuery = Helper.getSparqlUpdateQueryFromSpin(spinDeleteSetQueryN3, this.spinDeleteSetInsertQueryURI);
104 
105         // Replace all dependent variables within the queries
106         for (SparqlVariable sparqlVariable : semanticChange.getSparqlVariableList()) {
107             String name = sparqlVariable.getVariableName();
108             String value = sparqlVariable.getValue();
109             if (sparqlVariable.isResource()) {
110                 value = "<" + value + ">";
111             } else {
112                 value = "\"" + value + "\"";
113             }
114             dependencyMatchingQuery = dependencyMatchingQuery.replaceAll("\\?" + name, value);
115             addSetInsertQuery = addSetInsertQuery.replaceAll("\\?" + name, value);
116             deleteSetInsertQuery = deleteSetInsertQuery.replaceAll("\\?" + name, value);
117         }
118 
119         // Get the variables of the dependency matching query
120         sparqlVariablesList = new LinkedList<>();
121         HashMap<String, String> variableMapFromSpin = Helper.getVariableMapFromSpin(spinDependencyMatchingQueryN3, spinDependencyMatchingQueryURI);
122         for (String variableName : variableMapFromSpin.keySet()) {
123             String variableURI = variableMapFromSpin.get(variableName);
124             SparqlVariable sparqlVariable = new SparqlVariable(null, null, variableName, variableURI, null, false);
125             sparqlVariablesList.add(sparqlVariable);
126         }
127 
128     }
