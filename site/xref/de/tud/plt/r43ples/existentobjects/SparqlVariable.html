26      private String spinResourceURI;

38      /**
39       * The constructor.
40       *
41       * @param revisionGraph the revision graph
42       * @param sparqlVariableURI the SPARQL variable URI
43       * @throws InternalErrorException
44       */
45      public SparqlVariable(RevisionGraph revisionGraph, String sparqlVariableURI) throws InternalErrorException {
46          this.revisionGraph = revisionGraph;
47          this.revisionGraphURI = this.revisionGraph.getRevisionGraphUri();
48          this.sparqlVariableURI = sparqlVariableURI;
49  
50          retrieveAdditionalInformation();
51      }
52  
53      /**
54       * The constructor.
55       *
56       * @param revisionGraph the revision graph
57       * @param sparqlVariableURI the SPARQL variable URI
58       * @param variableName the variable name
59       * @param spinResourceURI the SPIN resource URI
60       * @param value the value
61       * @param isResource the identification whether the value is a resource (true) or a literal (false)
62       */
63      public SparqlVariable(RevisionGraph revisionGraph, String sparqlVariableURI, String variableName, String spinResourceURI, String value, boolean isResource) {
64          this.revisionGraph = revisionGraph;
65          // Null check is necessary because this constructor is also used with an empty revision graph as a temporary object
66          if (this.revisionGraph != null) {
67              this.revisionGraphURI = this.revisionGraph.getRevisionGraphUri();
68          }
69          this.sparqlVariableURI = sparqlVariableURI;
70          this.variableName = variableName;
71          this.spinResourceURI = spinResourceURI;
72          this.value = value;
73          this.isResource = isResource;
74      }
75  
76      /**
77       * The constructor. Used for the coevolution
78       *
79       * @param sparqlVariableURI the SPARQL variable URI
80       * @throws InternalErrorException
81       */
82      public SparqlVariable(String sparqlVariableURI) throws InternalErrorException {
83          this.revisionGraphURI = Config.evolution_graph;
84          this.sparqlVariableURI = sparqlVariableURI;
85  
86          retrieveAdditionalInformation();
87      }
88  
89      /**
90       * Calculate additional information of the current SPARQL variable and store this information to local variables.
91       *
92       * @throws InternalErrorException
93       */
94      private void retrieveAdditionalInformation() throws InternalErrorException {
95          logger.info("Get additional information of current SPARQL variable URI " + sparqlVariableURI + ".");
96          String query = Config.prefixes + String.format(""
97                  + "SELECT ?variableName ?spinResourceURI ?value "
98                  + "WHERE { GRAPH  <%s> {"
99                  + "	<%s> a aero:SPARQLVariable; "
100                 + "	 sp:varName ?variableName; "
101                 + "  aero:spinResource ?spinResourceURI; "
102                 + "  aero:value ?value. "
103                 + "} }", revisionGraphURI, sparqlVariableURI);
104         this.logger.debug(query);
105         ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
106         if (resultSet.hasNext()) {
107             QuerySolution qs = resultSet.next();
108             variableName = qs.getLiteral("?variableName").toString();
109             spinResourceURI = qs.getResource("?spinResourceURI").toString();
110 
111             if (qs.getResource("?value") != null) {
112                 value = qs.getResource("?value").toString();
113                 isResource = true;
114             } else {
115                 if (qs.getLiteral("?value") != null) {
116                     value = qs.getLiteral("?value").toString();
117                     isResource = false;
118                 }
119             }
120         } else {
121             throw new InternalErrorException("No additional information found for SPARQL variable URI " + sparqlVariableURI + ".");
122         }
123     }
