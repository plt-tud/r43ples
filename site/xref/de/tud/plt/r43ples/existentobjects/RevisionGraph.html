71  	 * 
72  	 * @return uri of the revision graph for this graph
73  	 */
74  	public String getRevisionGraphUri() {
75  		if (revisionGraphURI == null) {
76  			String query = String.format(
77  					"SELECT ?revisionGraph "
78  							+ "WHERE { GRAPH <%s> {"
79  							+ "	<%s> <http://eatld.et.tu-dresden.de/rmo#hasRevisionGraph> ?revisionGraph ."
80  							+ "} }", Config.revision_graph, graphName);
81  
82  			ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
83  
84  			if (results.hasNext()) {
85  				QuerySolution qs = results.next();
86  				revisionGraphURI = qs.getResource("?revisionGraph").toString();
87  			} else {
88  				return null;
89  			}
90  		}
91  
92  		return revisionGraphURI;
93  	}

109 						+ " { ?rev rmo:revisionIdentifier \"%s\"} UNION { ?ref rmo:referenceIdentifier \"%s\"} }} ",
110 						revisionGraph, identifier, identifier);
111 		return TripleStoreInterfaceSingleton.get().executeAskQuery(queryASK);

139 	public String getReferenceGraph(final String referenceIdentifier) throws InternalErrorException {
140 		String revisionGraph = this.getRevisionGraphUri();
141 		String query = Config.prefixes + String.format("" 
142 				+ "SELECT ?graph " 
143 				+ "WHERE { GRAPH  <%s> {" 
144 				+ "	?ref a rmo:Reference; "
145 				+ "		rmo:references ?rev;" 
146 				+ "		rmo:fullContent ?graph."
147 				+ " ?rev a rmo:Revision."
148 				+ "	{?ref rmo:referenceIdentifier \"%s\"} UNION {?rev rmo:revisionIdentifier \"%s\"}"
149 				+ "} }", revisionGraph, referenceIdentifier, referenceIdentifier);
150 		this.logger.debug(query);
151 		ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
152 		if (resultSet.hasNext()) {
153 			QuerySolution qs = resultSet.next();
154 			return qs.getResource("?graph").toString();
155 		} else {
156 			throw new InternalErrorException("No reference graph found for graph <" + graphName + "> and identifier \"" + referenceIdentifier + "\"");

170 		String queryString = Config.prefixes + String.format(""

176 		ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(queryString);

180 		}
181 		else {
182             throw new InternalErrorException("No master for graph <" + this.graphName + "> available");

187 	 * Returns new unique revision identifier for this graph.
188 	 *
189 	 * @return new revision identifier
190 	 * @throws InternalErrorException 
191 	 */
192 	public String getNextRevisionIdentifier() throws InternalErrorException {
193 		//
194 		int nextNumber;
195 		
196 		String revisionGraph = this.getRevisionGraphUri();
197 		String query = Config.prefixes
198 				+ String.format(
199 					"SELECT ?nr "
200 					+ "WHERE { GRAPH <%s> {"
201 					+ "	?rev a rmo:Revision; rmo:revisionIdentifier ?nr ."
202 					+ " } "
203 					+ "}ORDER BY DESC(xsd:integer(?nr))", revisionGraph);
204 		try {
205 			ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
206 			QuerySolution qs = results.next();
207 			nextNumber = qs.getLiteral("?nr").getInt()+1;
208 		}
209 		catch (Exception e){
210 			nextNumber = 0;
211 		}
212 		
213 		int count = 0;
214 		while (this.hasRevisionNumber(""+nextNumber)){
215 			nextNumber++;
216 			count++;
217 			if (count==100)
218 				throw new InternalErrorException("No new revision number found");
219 		}
220 		
221 		return ""+nextNumber;
222 	}

225 	 * Deletes all revision information for this revision graph including all full
226 	 * graphs and information in the R43ples system.
227 	 * 
228 	 */
229 	public void purgeRevisionInformation() {
230 		logger.info("Purge revision information of graph " + graphName);
231 		// Drop all full graphs as well as add and delete sets which are related to specified graph
232 		// If the revision graph is null there is nothing to delete
233 		String revisionGraph = this.getRevisionGraphUri();
234 		if (revisionGraph != null) {
235 			String query = Config.prefixes	+ String.format(""
236 					+ "SELECT DISTINCT ?graph "
237 					+ "WHERE { GRAPH <%s> {"
238 					+ " {?rev rmo:addSet ?graph}"
239 					+ " UNION {?rev rmo:deleteSet ?graph}"
240 					+ " UNION {?ref rmo:fullContent ?graph}"
241 					+ "} }", revisionGraph);
242 
243 			ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
244 			while (results.hasNext()) {
245 				QuerySolution qs = results.next();
246 				if (qs.get("?graph").isResource()) {
247 						String graph = qs.getResource("graph").toString();
248 						TripleStoreInterfaceSingleton.get().executeUpdateQuery("DROP SILENT GRAPH <" + graph + ">");
249 						logger.debug("Graph deleted: " + graph);
250 				}
251 			}
252 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format("DROP SILENT GRAPH <%s>", revisionGraph));
253 
254 			// Remove information from revision graph
255 			String queryDelete = Config.prefixes + String.format(
256 							"DELETE { "
257 							+ "GRAPH <%s> {	<%s> ?p ?o.}"
258 							+ "}"
259 							+ "WHERE {"
260 							+ "	GRAPH <%s> { <%s> a rmo:RevisionGraph; ?p ?o.}"
261 							+ "}"
262 							, Config.revision_graph, graphName, Config.revision_graph, graphName);
263 
264 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(queryDelete);
265 		}
266 	}

269 	 * get the names of all branches of the revision graph
270 	 * @return list of the name of branch
271 	 * */
272 	public ArrayList<String> getAllBranchNames() {
273 		logger.info("Get all branch names of graph "+ graphName);
274 		ArrayList<String> list = new ArrayList<String>();	
275 		if (graphName != null) {
276 			String revisionGraph = this.getRevisionGraphUri();
277 			String sparqlQuery = Config.prefixes
278 					+ String.format(
279 					  "SELECT DISTINCT ?label %n"
280 					+ "FROM <%s> %n"
281 					+ "WHERE { %n"
282 					+ "	?branch a rmo:Branch ;"
283 					+ "		rmo:referenceIdentifier ?label . "
284 					+ "} %n"
285 					+ "ORDER BY ?label", revisionGraph);
286 			
287 			ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(sparqlQuery);
288 			while (results.hasNext()) {
289 				QuerySolution qs = results.next();
290 				list.add(qs.getLiteral("label").toString());
291 			}		
292 		}
293 		logger.debug("All branches: " +list);
294 		return list;
295 	}

301 	 * @param referenceIdentifier
302 	 *            reference name or revision number
303 	 * @return URI of identified revision
304 	 * @throws InternalErrorException 
305 	 */
306 	public String getBranchUri(final String referenceIdentifier) throws InternalErrorException {
307 		String revisionGraph = this.getRevisionGraphUri();
308 		String query = Config.prefixes
309 				+ String.format("SELECT ?ref " 
310 						+ "WHERE { GRAPH <%s> {"
311 						+ "	?ref a rmo:Branch; rmo:references ?rev."
312 						+ " ?rev a rmo:Revision."
313 						+ "	{?rev rmo:revisionIdentifier \"%s\".} UNION {?ref rmo:referenceIdentifier \"%s\" .}"
314 						+ "} }",
315 						revisionGraph, referenceIdentifier, referenceIdentifier);
316 		ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
317 		if (resultSet.hasNext()) {
318 			QuerySolution qs = resultSet.next();
319 			if (resultSet.hasNext()) {
320 				throw new InternalErrorException("Identifier is not unique for specified graph name: "
321 						+ referenceIdentifier);
322 			}
323 			return qs.getResource("?ref").toString();
324 		} else {
325 			throw new InternalErrorException("No Revision or Reference found with identifier: "
326 					+ referenceIdentifier);
327 		}

343 	 * Get the revision URI for a given reference name or revision number
344 	 * 
345 	 * @param revisionIdentifier
346 	 *            reference name or revision number
347 	 * @return URI of identified revision
348 	 * @throws InternalErrorException 
349 	 */
350 	public String getRevisionUri(final String revisionIdentifier) throws InternalErrorException {
351 		String revisionGraph = this.getRevisionGraphUri();
352 		String query = Config.prefixes
353 				+ String.format(
354 						"SELECT ?rev WHERE { GRAPH <%s> {"
355 							+ "{?rev a rmo:Revision; rmo:revisionIdentifier \"%s\" .}"
356 							+ "UNION {?rev a rmo:Revision. ?ref a rmo:Reference; rmo:references ?rev; rmo:referenceIdentifier \"%s\" .}"
357 							+ "} }", revisionGraph, revisionIdentifier, revisionIdentifier);
358 		ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
359 		if (resultSet.hasNext()) {
360 			QuerySolution qs = resultSet.next();
361 			if (resultSet.hasNext()) {
362 				logger.error("Identifier not unique: " + revisionIdentifier);
363 				throw new InternalErrorException("Identifier not unique: " + revisionIdentifier);
364 			}
365 			return qs.getResource("?rev").toString();
366 		} else {
367 			logger.error("No Revision or Reference found with identifier: " + revisionIdentifier);
368 			throw new InternalErrorException("No Revision or Reference found with identifier: "
369 					+ revisionIdentifier);
370 		}
371 	}

441 						revisionGraph, revisionNumber);
442 		return TripleStoreInterfaceSingleton.get().executeAskQuery(queryASK);
