47  	private static Logger logger = LogManager.getLogger(R43plesService.class);

52  	/**
53  	 * Starts the server.
54  	 * 
55  	 * @param args
56  	 * @throws ConfigurationException
57  	 * @throws URISyntaxException
58  	 * @throws IOException 
59  	 */
60  	public static void main(String[] args) {
61  		
62  		
63  		// command-line-parser: JCommander-1.29
64  		R43plesArgs jci = new R43plesArgs();
65  		JCommander jc = new JCommander(jci);
66  
67  		// wrong input => display usage
68  		try {
69  			jc.parse(args);
70  		} catch (ParameterException e) {
71  			jc.usage();
72  			System.err.println(e.toString());
73  			System.exit(0);
74  		}
75  		
76  		if (jci.help) {
77  			jc.usage();
78  			System.exit(0);
79  		}
80  		
81  		logger.debug("config: " + jci.config);
82  		Runtime.getRuntime().addShutdownHook(new Thread() {
83  			public void run() { R43plesService.stop(); }
84  		});
85  		try {
86  			Config.readConfig(jci.config);
87  			start();
88  		} catch (Exception e) {
89  			e.printStackTrace();
90  			System.exit(1);
91  		}
92  		logger.info("Press ctrl+c to quit the server");
93  		while (true);		
94  	}

96  	
97  	/**
98  	 * Starts the server. It is possible to enable a secure connection.
99  	 * 
100 	 * @throws ConfigurationException
101 	 * @throws URISyntaxException
102 	 * @throws IOException 
103 	 */
104 	public static void start() throws ConfigurationException, URISyntaxException, IOException {		
105 		logger.info("Starting R43ples on grizzly...");
106 		URI BASE_URI;
107 		
108 		ResourceConfig rc = new ResourceConfig()
109 			.registerClasses(Endpoint.class, 
110 					Misc.class,
111 					API.class, 
112 					Configuration.class, 
113 					Debug.class, 
114 					Merging.class)
115 			.property(MustacheMvcFeature.TEMPLATE_BASE_PATH, "templates")
116 			.register(MustacheMvcFeature.class)
117 			.register(ExceptionMapper.class)
118 			.register(JacksonFeature.class);
119 		
120 		SSLContextConfigurator sslCon =  new SSLContextConfigurator();
121 		
122 		// Try to initialize SSL keys
123 		if (Config.ssl_keystore != null && Config.ssl_password != null) {
124 			sslCon.setKeyStoreFile(Config.ssl_keystore);
125 			sslCon.setKeyStorePass(Config.ssl_password);
126 			sslCon.setTrustStoreFile(Config.ssl_keystore);
127 			sslCon.setTrustStorePass(Config.ssl_password);
128 		}
129 		
130 		if (sslCon.validateConfiguration()) {
131 			logger.info("SSL context validated");
132 			BASE_URI = new URI("https", null, Config.service_host, Config.service_port, Config.service_path, null, null);
133 			server = GrizzlyHttpServerFactory.createHttpServer(BASE_URI, rc, true, new SSLEngineConfigurator(sslCon, false, false, false));		
134 		} else {
135 			logger.info("SSL context not validated");
136 			BASE_URI = new URI("http", null, Config.service_host, Config.service_port, Config.service_path, null, null);
137 			server = GrizzlyHttpServerFactory.createHttpServer(BASE_URI, rc);			
138 		}
139 		
140 		for (NetworkListener l : server.getListeners()) { l.getFileCache().setEnabled(false); }
141 		logger.info("File cache disabled");
142 		
143 		server.getServerConfiguration().addHttpHandler(
144 		        new CLStaticHttpHandler(R43plesService.class.getClassLoader(),"webapp/"), "/static/");
145 
146 		server.start();
147 
148 		logger.info(String.format("Server started - R43ples endpoint available under: %s/sparql", BASE_URI));
149 		
150 		String version = R43plesService.class.getPackage().getImplementationVersion();
151 		if (version==null){
152 			try{
153 				version = "Commit: " +GitRepositoryState.getGitRepositoryState().commitIdAbbrev;
154 			}
155 			catch(Exception e){
156                 version="No version information available";
157             }
158 		}
159 		logger.info("Version: "+ version);
160 	}

163 	/**
164 	 * Stops the server.
165 	 */
166 	public static void stop() {
167 		logger.info("Server shutdown ...");
168 		TripleStoreInterfaceSingleton.close();
169 		server.shutdown();
170 	}
