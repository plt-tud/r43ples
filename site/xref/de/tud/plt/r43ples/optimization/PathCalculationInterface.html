3   import de.tud.plt.r43ples.exception.InternalErrorException;
4   import de.tud.plt.r43ples.existentobjects.Path;
5   import de.tud.plt.r43ples.existentobjects.Revision;
6   
7   
8   /**
9    * This interface provides methods to calculate paths between revisions.
10   *
11   * @author Stephan Hensel
12   *
13   */
14  public interface PathCalculationInterface {
15  
16  
17      /**
18       * Get the path to the nearest revision which has a full graph.
19       *
20       * @param revision      revision where the search should start
21       * @return path containing all revisions from start revision to next revision with a full graph
22       */
23      Path getPathToRevisionWithFullGraph(Revision revision) throws InternalErrorException;
24  
25  
26      /**
27       * Get the common revision of the specified revisions which has the shortest path to the two.
28       * To ensure wise results the revisions should be terminal branch nodes.
29       *
30       * @param revision1     the first revision should be a terminal branch node
31       * @param revision2     the second revision should be a terminal branch node
32       * @return the nearest common revision
33       * @throws InternalErrorException
34       */
35      Revision getCommonRevisionWithShortestPath(Revision revision1, Revision revision2) throws InternalErrorException;
36  
37      /**
38       * Calculate the path from start revision to target revision.
39       * Example: target rmo:wasDerivedFrom source
40       *
41       * @param startRevision  the start revision
42       * @param targetRevision the target revision
43       * @return path containing all revisions from start revision to target revision
44       * @throws InternalErrorException
45       */
46      Path getPathBetweenStartAndTargetRevision(Revision startRevision, Revision targetRevision) throws InternalErrorException;
47  
48  
49      ChangeSetPath getChangeSetsBetweenStartAndTargetRevision(Revision startRevision, Revision targetRevision) throws InternalErrorException;
50  
51      ChangeSetPath getPathOfChangeSets(Revision revision) throws InternalErrorException;
52  
53  }
