48  	 * Generates difference between revision a and revision b of a named graph.
49  	 * Revision a is considered older than revision b (a -> b)
50  	 * @param graphName name of graph
51  	 * @param revAIdentifier identifier revision a
52  	 * @param revBIdentifier identifier revision b
53  	 * @param fileFormat format of outputfile. possible options: nquads, trig
54  	 * @return list of diffs
55  	 */
56  	public static String getDiffsBetweenStartAndTargetRevision(
57  			final String graphName,
58  			final String revAIdentifier,
59  			final String revBIdentifier,
60  			final String fileFormat) {
61  
62  		// create revision graph
63  		RevisionGraph graph = new RevisionGraph(graphName);
64  		PathCalculationInterface calculation = PathCalculationFabric.getInstance(graph);
65  
66  		// get revisions
67  		Revision revA = null;
68  		Revision revB = null;
69  
70  		try {
71  			revA = new Revision(graph, revAIdentifier, true);
72  		} catch (InternalErrorException e) {
73  			e.printStackTrace();
74  			return "Error finding revision A:\n" + e.getMessage();
75  		}
76  
77  		try {
78  			revB = new Revision(graph, revBIdentifier, true);
79  		} catch (InternalErrorException e) {
80  			e.printStackTrace();
81  			return "Error finding revision B:\n" + e.getMessage();
82  		}
83  
84  		// get changeset path
85  		ChangeSetPath path = null;
86  		try {
87  			path = calculation.getChangeSetsBetweenStartAndTargetRevision(revA, revB);
88  		} catch (InternalErrorException e) {
89  			e.printStackTrace();
90  			return "Error creating changeset. Make sure revision a lies before revision b:\n" + e.getMessage();
91  		}
92  
93  		// process changeset to get diffs
94  
95  		LinkedList<ChangeSet> changeSets = path.getRevisionPath();
96  
97  		// first changeset as base
98  		// ! changeSets starts with last changeset (inverse)
99  		ChangeSet oldCS = changeSets.get(changeSets.size() - 1);
100 
101 		Model oldAddModel = JenaModelManagement.readStringToJenaModel(oldCS.getAddSetContent(), "TURTLE");
102 		Model oldDeleteModel = JenaModelManagement.readStringToJenaModel(oldCS.getDeleteSetContent(), "TURTLE");
103 
104 		// generate diff model (cross compare add and delete sets)
105 		for (int i = changeSets.size() - 2; i >= 0; i--) {
106 
107 			ChangeSet newCS = changeSets.get(i);
108 
109 			Model addModel = JenaModelManagement.readStringToJenaModel(newCS.getAddSetContent(), "TURTLE");
110 			Model deleteModel = JenaModelManagement.readStringToJenaModel(newCS.getDeleteSetContent(), "TURTLE");
111 
112 			// find neutralizing triples and delete them
113 			Model neutralAdd = oldDeleteModel.intersection(addModel);
114 			Model neutralDelete = oldAddModel.intersection(deleteModel);
115 
116 			Model test = oldDeleteModel.remove(neutralAdd);
117 			addModel.remove(neutralAdd);
118 			oldAddModel.remove(neutralDelete);
119 			deleteModel.remove(neutralDelete);
120 
121 			// add remaining triples of new model to old model
122 			oldDeleteModel.add(deleteModel);
123 			oldAddModel.add(addModel);
124 		}
125 
126 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
127 
128 		// create dataset from add and delete model for transmission
129 		Dataset resultSet = DatasetFactory.create();
130 		resultSet.addNamedModel("AddSet", oldAddModel);
131 		resultSet.addNamedModel("DeleteSet", oldDeleteModel);
132 
133 		// create returnable result string
134 		if (fileFormat.toLowerCase().contains("trig"))
135 			RDFDataMgr.write(baos, resultSet, RDFFormat.TRIG_PRETTY);
136 		else if (fileFormat.toLowerCase().contains("nquads"))
137 			RDFDataMgr.write(baos, resultSet, RDFFormat.NQUADS_UTF8);
138 		else {
139 			return "unsupported file format";
140 		}
141 
142 		return baos.toString();
143 	}

177 	public static void splitAndExecuteBigQuery(final String graphName, final String dataSetAsNTriples, final String template){
178 		final int MAX_STATEMENTS = 500;
179 		String[] lines = dataSetAsNTriples.split("\n");
180 		int counter = 0;
181 		StringBuilder insert = new StringBuilder();
182 		
183 		for (int i=0; i < lines.length; i++) {
184 			insert.append(lines[i]);
185 			insert.append("\n");
186 			counter++;
187 			if (counter == MAX_STATEMENTS-1) {
188 				TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(template, graphName, insert));
189 				counter = 0;
190 				insert = new StringBuilder();
191 			}
192 		}
193 
194 		TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(template, graphName, insert));
195 	}

244 	 * Get the user URI. If the user does not exist then the user will be created.
245 	 *
246 	 * @param user name as string
247 	 * @return URI of user
248 	 */
249 	public static String getUserURI(final String user) {
250 		// When user does not already exists - create new
251 
252 		String query = Config.prefixes
253 				+ String.format("SELECT ?personUri { GRAPH <%s>  { " + "?personUri a prov:Person;"
254 						+ "  rdfs:label \"%s\"." + "} }", Config.revision_graph, user); //TODO check if the users are created within the special revision graph - maybe move to RevisionGraph
255 		ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
256 		if (results.hasNext()) {
257 			logger.debug("User " + user + " already exists.");
258 			QuerySolution qs = results.next();
259 			return qs.getResource("?personUri").toString();
260 		} else {
261 			String personUri = null;
262 			try {
263 				personUri = "http://eatld.et.tu-dresden.de/persons/" + URLEncoder.encode(user, "UTF-8");
264 			} catch (UnsupportedEncodingException e) {
265 				e.printStackTrace();
266 			}
267 			logger.debug("User does not exists. Create user " + personUri + ".");
268 			query = Config.prefixes
269 					+ String.format("INSERT DATA { GRAPH <%s> { <%s> a prov:Person; rdfs:label \"%s\". } }",
270 							Config.revision_graph, personUri, user);
271 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(query);
272 			return personUri;
273 		}
274 	}

278 	 * Get a SPARQL SELECT query in human readable syntax from SPIN.
279 	 *
280 	 * @param spinQueryN3 the SPIN content as N3
281 	 * @param resourceURI the resource URI of the SELECT query to search for
282 	 * @return the SPARQL query in human readable syntax
283 	 */
284 	public static String getSparqlSelectQueryFromSpin(String spinQueryN3, String resourceURI) {
285 
286 		Model model = JenaModelManagement.readNTripleStringToJenaModel(spinQueryN3);
287 
288 		StmtIterator it = model.listStatements(null, RDF.type, SP.Select);
289 		while(it.hasNext()) {
290 			Resource rsrc = it.next().getSubject();
291 			if (rsrc.toString().equals(resourceURI)) {
292 				org.topbraid.spin.model.Query newQuery = SPINFactory.asQuery(rsrc);
293 				return ARQFactory.get().createQuery(newQuery).toString();
294 			}
295 		}
296 
297 		return null;
298 
299 	}

326 	 * Get a map of SPARQL variables and there URIs in SPIN.
327 	 *
328 	 * @param spinQueryN3 the SPIN content as N3
329 	 * @param resourceURI the resource URI of the SELECT query to search for
330 	 * @return the map
331 	 */
332 	public static HashMap<String,String> getVariableMapFromSpin(String spinQueryN3, String resourceURI) {
333 
334 		HashMap<String,String> resultMap = new HashMap<>();
335 
336 		Model model = JenaModelManagement.readNTripleStringToJenaModel(spinQueryN3);
337 
338 		// Create SELECT query to query further resources from current URI
339 		String selectQuery = String.format(
340 				  "SELECT ?variables %n"
341 				+ "WHERE { %n"
342 				+ "	 <%s> <http://spinrdf.org/sp#resultVariables> ?variables . %n"
343 				+ "} %n", resourceURI);
344 
345 		Query qry = QueryFactory.create(selectQuery);
346 		QueryExecution qe = QueryExecutionFactory.create(qry, model);
347 		ResultSet resultSet = qe.execSelect();
348 		while (resultSet.hasNext()) {
349 			QuerySolution qs = resultSet.next();
350 			Resource resource;
351 			try {
352 				resource = qs.getResource("?variables");
353 			} catch (Exception e) {
354 				resource = null;
355 			}
356 
357 			if (resource != null) {
358 				String startURI = resource.toString();
359 				resultMap = calculateVariableMapFromSpin(model, startURI);
360 			}
361 		}
362 
363 		return resultMap;
364 
365 	}

367 	/**
368 	 * Get a map of SPARQL variables and there URIs in SPIN.
369 	 *
370 	 * @param model the jena model
371 	 * @param startURI the resource URI of the SELECT query to search for
372 	 * @return the map
373 	 */
374 	private static HashMap<String,String> calculateVariableMapFromSpin(Model model, String startURI) {
375 
376 		HashMap<String, String> resultMap = new HashMap<>();
377 
378 		// Get first of given resource
379 		String selectQuery = String.format(
380 				"SELECT ?first ?rest ?varName %n"
381 						+ "WHERE { %n"
382 						+ "	 <%s> <http://www.w3.org/1999/02/22-rdf-syntax-ns#first> ?first . %n"
383 						+ "	 ?first <http://spinrdf.org/sp#varName> ?varName . %n"
384 						+ "	 <%s> <http://www.w3.org/1999/02/22-rdf-syntax-ns#rest> ?rest . %n"
385 						+ "} %n", startURI, startURI);
386 
387 		Query qry = QueryFactory.create(selectQuery);
388 		QueryExecution qe = QueryExecutionFactory.create(qry, model);
389 		ResultSet resultSet = qe.execSelect();
390 		while (resultSet.hasNext()) {
391 			QuerySolution qs = resultSet.next();
392 			Resource resourceFirst = qs.getResource("?first");
393 			Literal literalVarName = qs.getLiteral("?varName");
394 			Resource resourceRest = qs.getResource("?rest");
395 
396 			if ((resourceFirst != null) && (literalVarName != null)) {
397 				resultMap.put(literalVarName.toString(), resourceFirst.toString());
398 			}
399 			// Get rest
400 			if (resourceRest != null) {
401 				if (!resourceRest.toString().equals("http://www.w3.org/1999/02/22-rdf-syntax-ns#nil")) {
402 					resultMap.putAll(calculateVariableMapFromSpin(model, resourceRest.toString()));
403 				}
404 			}
405 		}
406 
407 		return resultMap;
408 	}

411 	 * Get all related elements to a start URI.
412 	 *
413 	 * @param graph the graph to search in
414 	 * @param startURI the start URI
415 	 * @return the related elements
416 	 */
417 	public static String getAllRelatedElementsToURI(String graph, String startURI) {
418 
419 		LinkedList<String> elements = calculateListOfURIsRelatedToStartURI(graph, startURI);
420 
421 		String constructQueryTemplate =
422 				"CONSTRUCT {?s ?p ?o} %n"
423 						+ "WHERE { GRAPH <%s> { %n"
424 						+ " ?s ?p ?o . %n"
425 						+ "}"
426 						+ "FILTER (?s = <%s>) "
427 						+ "} %n";
428 
429 		StringBuilder sb = new StringBuilder();
430 
431 		// Iterate through the detected URIs and make construct queries
432 		for (String currentURI : elements) {
433 			String constructQuery = String.format(constructQueryTemplate, graph, currentURI);
434 			sb.append(TripleStoreInterfaceSingleton.get().executeConstructQuery(constructQuery, FileUtils.langNTriple));
435 		}
436 
437 		return sb.toString();
438 
439 	}

441 	/**
442 	 * Calculates a list of URIs related to a given start URI.
443 	 *
444 	 * @param graph the graph to search in
445 	 * @param startURI the start URI
446 	 * @return the list of related URIs
447 	 */
448 	private static LinkedList<String> calculateListOfURIsRelatedToStartURI(String graph, String startURI) {
449 
450 		LinkedList<String> elements = new LinkedList<>();
451 
452 		String selectQueryTemplate =
453 				"SELECT ?p ?o %n"
454 						+ "WHERE { GRAPH <%s> { %n"
455 						+ "	<%s> ?p ?o . %n"
456 						+ "} } %n";
457 
458 		// Add the current URI to the list of already processed URIs
459 		elements.add(startURI);
460 
461 		// Create SELECT query to query further resources from current URI
462 		String selectQuery = String.format(selectQueryTemplate, graph, startURI);
463 
464 		// Iterate over results
465 		ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(selectQuery);
466 		while (resultSet.hasNext()) {
467 			QuerySolution qs = resultSet.next();
468 			Resource resource;
469 			try {
470 				resource = qs.getResource("?o");
471 			} catch (Exception e) {
472 				resource = null;
473 			}
474 
475 			if (resource != null) {
476 				String resourceString = resource.toString();
477 				if (!elements.contains(resourceString)) {
478 					elements.addAll(calculateListOfURIsRelatedToStartURI(graph, resourceString));
479 				}
480 			}
481 		}
482 
483 		return elements;
484 
485 	}
