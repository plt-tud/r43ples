3   import java.net.URI;
4   import java.util.HashMap;
5   import java.util.Map;
6   
7   import javax.ws.rs.FormParam;
8   import javax.ws.rs.GET;
9   import javax.ws.rs.POST;
10  import javax.ws.rs.Path;
11  import javax.ws.rs.QueryParam;
12  import javax.ws.rs.core.Context;
13  import javax.ws.rs.core.Response;
14  import javax.ws.rs.core.UriInfo;
15  
16  import org.apache.logging.log4j.LogManager;
17  import org.apache.logging.log4j.Logger;
18  import org.glassfish.jersey.server.mvc.Template;
19  
20  import de.tud.plt.r43ples.exception.InternalErrorException;
21  import de.tud.plt.r43ples.management.Config;
22  /**
23   * REST services for managing the configuration on R43ples
24   * @author mgraube
25   *
26   */
27  @Path("configuration")
28  public class Configuration {
29  	
30  	@Context UriInfo uriInfo;
31  	
32  	private final static Logger logger = LogManager.getLogger(Configuration.class);
33  	
34  	/**
35  	 * Get all configuration variables of R43ples and the user defined namespaces
36  	 * @return map with all configuration variables
37  	 * @throws InternalErrorException
38  	 */
39  	@GET
40  	@Template(name = "/configuration.mustache")
41  	public final Map<String, Object> getConfiguration() throws InternalErrorException {
42  		logger.info("Get Configuration page");
43  		Map<String, Object> htmlMap = new HashMap<String, Object>();
44  		htmlMap.put("version", Endpoint.class.getPackage().getImplementationVersion());
45  	    htmlMap.put("revisionGraph", Config.revision_graph);
46  	    htmlMap.put("triplestore_type", Config.triplestore_type);
47  	    htmlMap.put("triplestore_url", Config.triplestore_url);
48  	    htmlMap.put("sdg_graph", Config.sdg_graph);
49  	    htmlMap.put("namespaces", Config.user_defined_prefixes.entrySet());
50  	    htmlMap.put("configuration_active", true);
51  		return htmlMap;
52  	}
53  	
54  	/**
55  	 * Add user defined prefix to R43ples configuration
56  	 * 
57  	 * @param prefix	prefix to be added
58  	 * @param namespace	namespaces used for specified prefix
59  	 * @return	updated configuration
60  	 * @throws InternalErrorException
61  	 */
62  	@POST
63  	@Template(name = "/configuration.mustache")
64  	public final Map<String, Object> addPrefix(
65  			@FormParam("prefix") final String prefix,
66  			@FormParam("namespace") final String namespace) throws InternalErrorException {
67  		if (prefix!=null && namespace!=null){
68  			Config.user_defined_prefixes.put(prefix, namespace);
69  		}
70  		return getConfiguration();
71  	}
72  	
73  	/**
74  	 * Delete specified prefix from R43ples configuration
75  	 * @param prefix	prefix which specified the namespace to be deleted
76  	 * @return redirect to configuration site
77  	 * @throws InternalErrorException
78  	 */
79  	@GET
80  	@Path("deletePrefix")
81  	public final Response deletePrefix(
82  			@QueryParam("prefix") final String prefix) throws InternalErrorException {
83  		if (prefix!=null && Config.user_defined_prefixes.containsKey(prefix)){
84  			Config.user_defined_prefixes.remove(prefix);
85  		}
86  		URI uri = uriInfo.getBaseUriBuilder().path("configuration").build();
87  		return Response.seeOther(uri).build();
88  	}
89  	
90  
91  }
