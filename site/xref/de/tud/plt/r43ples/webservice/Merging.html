20  @Path("merging")
21  public class Merging {
22  	
23  	private final static Logger logger = LogManager.getLogger(Merging.class);
24  
25  	/**
26  	 * get merging HTML start page and input merging information
27  	 * */
28  	@GET
29  	public final Response getMerging() {
30  		logger.info("Merging - Start page");		
31  		
32  		List<String> graphList = new LinkedList<>();
33  		graphList.addAll(new RevisionControl().getRevisedGraphs().keySet());
34  		Map<String, Object> scope = new HashMap<String, Object>();
35  	    scope.put("merging_active", true);
36  		scope.put("graphList", graphList);
37  		
38  		StringWriter sw = new StringWriter();
39  		MustacheFactory mf = new DefaultMustacheFactory();
40  		Mustache mustache = mf.compile("templates/merge_start_wip.mustache");
41  		mustache.execute(sw, scope);		
42  		return Response.ok().entity(sw.toString()).type(MediaType.TEXT_HTML).build();
43  	}
