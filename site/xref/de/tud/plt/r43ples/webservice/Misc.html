133 	/**
134 	 * Provide revision information about R43ples system.
135 	 * 
136 	 * @param graphName
137 	 *            Provide only information about this graph
138 	 * @return RDF model of revision information
139 	 */
140 	@Path("revisiongraph")
141 	@GET
142 	@Produces({ TEXT_TURTLE, APPLICATION_RDF_XML, MediaType.APPLICATION_JSON, MediaType.TEXT_HTML })
143 	public final Response getRevisionGraph(@HeaderParam("Accept") final String format_header,
144 			@QueryParam("format") final String format_query,
145 			@QueryParam("graph")  final String graphName) throws Exception {
146 		String format = null;
147 		if (format_query == null){
148 			List<Variant> reqVariants = Variant.mediaTypes(TEXT_TURTLE_TYPE,
149 					APPLICATION_RDF_XML_TYPE, MediaType.APPLICATION_JSON_TYPE).build();
150 			Variant bestVariant = request.selectVariant(reqVariants);
151 			if (bestVariant == null) {
152 				throw new Exception("Requested datatype not available");
153 			}
154 			MediaType reqMediaType = bestVariant.getMediaType();
155 			format = reqMediaType.toString();
156 		}
157 		else {
158 			format = format_query;
159 		}
160 		logger.info("Get Revision Graph: " + graphName + " (format: " + format+")");
161 		
162 		ResponseBuilder response = Response.ok();
163 		if ("table".equalsIgnoreCase(format)) {
164 			response.type(MediaType.TEXT_HTML);
165 			response.entity(VisualisationTable.getHtmlOutput(graphName));
166 		}  else if ("graph".equalsIgnoreCase(format)) {
167 			response.type(MediaType.TEXT_HTML);
168 			response.entity(VisualisationGraph.getHtmlOutput(graphName));
169 		}
170 		else {
171 			RevisionGraph graph = new RevisionGraph(graphName);
172 			response.entity(graph.getContentOfRevisionGraph(format));
173 			response.type(format);
174 		}
175 		return response.build();
176 	}
