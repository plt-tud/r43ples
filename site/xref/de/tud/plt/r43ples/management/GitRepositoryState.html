4   import org.apache.commons.configuration.PropertiesConfiguration;

6   public class GitRepositoryState {
7   
8   	private static GitRepositoryState gitRepositoryState = null;
9   
10  	public String branch; // =${git.branch}
11  	public String describe; // =${git.commit.id.describe}
12  	public String describeShort; // =${git.commit.id.describe-short}
13  	public String commitId; // =${git.commit.id}
14  	public String commitIdAbbrev; // =${git.commit.id.abbrev}
15  	public String commitTime; // =${git.commit.time}
16  
17  	private GitRepositoryState(PropertiesConfiguration properties) {
18  		this.branch = properties.getString("git.branch");
19  		this.describe = properties.getString("git.commit.id.describe");
20  		this.describeShort = properties.getString("git.commit.id.describe-short");
21  		this.commitId = properties.getString("git.commit.id");
22  		this.commitIdAbbrev = properties.getString("git.commit.id.abbrev");
23  		this.commitTime = properties.getString("git.commit.time");
24  	}
25  
26  	public static GitRepositoryState getGitRepositoryState() {
27  
28  		if (gitRepositoryState == null) {
29  			PropertiesConfiguration config;
30  			try {
31  				config = new PropertiesConfiguration("git.properties");
32  				gitRepositoryState = new GitRepositoryState(config);
33  			} catch (ConfigurationException e) {
34  				e.printStackTrace();
35  			}
36  
37  		}
38  		return gitRepositoryState;
39  	}
