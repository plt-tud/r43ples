6   public class OutdatedException extends InternalErrorException{
7   
8       /** The default serial version UID **/
9       private static final long serialVersionUID = 1L;
10      private final Model clientModel;
11      private final Model serverModel;
12  
13  
14      public OutdatedException(Model clientModel, Model serverModel) {
15          super(String.format("Client information is not up to date. The following statements have changed: \n" +
16                          "Added:\n %s\n" +
17                          "Deleted:\n %s\n",
18                  JenaModelManagement.convertJenaModelToNTriple(serverModel.difference(clientModel)),
19                  JenaModelManagement.convertJenaModelToNTriple(clientModel.difference(serverModel))));
20          this.clientModel = clientModel;
21          this.serverModel = serverModel;
22          }
23  }
