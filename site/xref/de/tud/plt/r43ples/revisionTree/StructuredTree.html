37  	public static StructuredTree getTreeOfGraph(String revisionGraph) {
38  		// new Tree
39  		StructuredTree t = new StructuredTree();
40  		t.updateCommits(revisionGraph);
41  		t.updateBranches(revisionGraph);
42  		t.updateTags(revisionGraph);
43  
44  		// link commits to branches
45  		for(Commit c : t.commits) {
46  			//get branch of commit
47  			if (commit_branch_tmp.containsKey(c)) {
48  				Branch b = t.branches.get(t.branches.indexOf(new Branch(commit_branch_tmp.get(c))));
49  				c.setBranch(b);
50  			}
51  			else
52  				c.setBranch(t.branches.get(0));
53  		}
54  		return t;
55  	}

57  	private void updateBranches(String revisionGraph) {
58  		//query all branches
59  		
60  		String queryBranches = Config.prefixes + String.format(
61  						"SELECT ?branch ?title ?commit%n"
62  						+ "WHERE { GRAPH <%s> { %n"
63  						+ "  ?branch a rmo:Branch;%n"
64  						+ "     rmo:referenceIdentifier ?title;%n"
65  						+ "     rmo:references ?rev."
66  						+ "  ?commit a rmo:UpdateCommit;%n"
67  						+ "     rmo:generated ?rev.%n"
68  						+ "} }", revisionGraph);
69  
70  		ResultSet resultsBranches = TripleStoreInterfaceSingleton.get().executeSelectQuery(queryBranches);
71  
72  		while (resultsBranches.hasNext()) {
73  			QuerySolution sol = resultsBranches.next();
74  			Branch b = new Branch(sol.get("branch").toString(), sol.getLiteral("title").toString(), commits.get(commits
75  					.indexOf(new Commit(sol.get("commit").toString()))));
76  			branches.add(b);
77  		}
78  	}

80  	private void updateTags(String revisionGraph) {
81  		// query all tags
82  		String queryBranches = Config.prefixes + String.format(
83  						"SELECT ?tag ?title ?commit%n"
84  						+ "WHERE { GRAPH <%s> {%n"
85  						+ "?tag a rmo:Tag;%n"
86  						+ "rmo:referenceIdentifier ?title;%n"
87  						+ "rmo:references ?rev."
88  						+ "?commit a rmo:TagCommit;%n"
89  						+ "rmo:generated ?rev.%n"
90  						+ "} }", revisionGraph);
91  
92  		ResultSet resultsTags = TripleStoreInterfaceSingleton.get().executeSelectQuery(queryBranches);
93  
94  		while (resultsTags.hasNext()) {
95  			QuerySolution sol = resultsTags.next();
96  			Tag t = new Tag(sol.get("tag").toString(), sol.getLiteral("title").toString(), commits.get(commits
97  					.indexOf(new Commit(sol.get("commit").toString()))));
98  			tags.add(t);
99  		}
100 	}

117 	private void updateCommits(String revisionGraph) {
118 		// query all commits
119 		String queryCommits = Config.prefixes + String.format(
120 						"SELECT ?commit ?time ?prev ?next ?title ?authname ?branch%n"
121 						+ "WHERE { GRAPH <%s> {%n"
122 						+ "?commit a rmo:UpdateCommit;%n"
123 						+ "rmo:commitMessage ?title;%n"
124 						+ "rmo:used ?reva;%n"
125 						+ "rmo:generated ?revb;%n"
126 						+ "rmo:timeStamp ?time;%n"
127 						+ "rmo:wasAssociatedWith ?author.%n"
128 						+ "OPTIONAL { ?author rdfs:label ?authname. }%n"
129 						+ "?reva rmo:revisionIdentifier ?prev.%n"
130 						+ "?revb rmo:revisionIdentifier ?next."
131 						+ "OPTIONAL { ?revb rmo:belongsTo ?branch. }%n"
132 						+ "} }",//TODO belongsTo does not exist any more
133 						revisionGraph);
134 		
135 		ResultSet resultsCommits = TripleStoreInterfaceSingleton.get().executeSelectQuery(queryCommits);
136 
137 		// generate list of commits
138 		DateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
139 		while (resultsCommits.hasNext()) {
140 			QuerySolution sol = resultsCommits.next();
141 			try {
142 				Commit commit = new Commit(sol.get("commit").toString());
143 				// commit can be multiple times in solution set because of
144 				// multiple "prov:used" attributes
145 				if (commits.contains(commit))
146 				{
147 					// if commit is already in list, just add new base revision
148 					// to existing commit
149 					commits.get(commits.indexOf(commit)).getBaseRevisions().add(sol.getLiteral("prev").getString());
150 				} else {
151 					// generate new commit object
152 					String authname = sol.getLiteral("authname") == null ? "" : sol.getLiteral("authname").getString();
153 					Commit newCommit = new Commit(
154 							sol.get("commit").toString(),
155 							sol.getLiteral("title").getString(),
156 							df.parse(sol.getLiteral("time").toString()),
157 							authname,
158 							sol.getLiteral("prev").getString(),
159 							sol.getLiteral("next").getString());
160 					// save branch uri in Map to set branch reference later
161 					// after branches were generated
162 					if (sol.get("branch") != null)
163 						commit_branch_tmp.put(newCommit, sol.get("branch").toString());
164 					commits.add(newCommit);
165 				}
166 			} catch (ParseException e) {
167 				logger.error("Commit could not be parsed!");
168 			}
169 		}
170 
171 		// fill predecessor and successor lists of commits
172 		for (Commit c : commits) {
173 			for (Commit b : commits) {
174 				if (c.getBaseRevisions().contains(b.getNextRevision()))
175 					c.predecessor.add(b);
176 				else if (b.getBaseRevisions().contains(c.getNextRevision()))
177 					c.successors.add(b);
178 			}
179 		}
180 
181 		// sort list of commits by order of their occurrence
182 		Collections.sort(commits);
183 	}
