20  public class EndpointTest extends R43plesTest {
21      private static DataSetGenerationResult ds1;
22      private static DataSetGenerationResult ds2;
23  
24      private static String get_query_template(String rev1, String rev2) {
25          return String.format(""
26                          + "PREFIX : <http://test.com/> %n"
27                          + "SELECT ?address %n"
28                          + "WHERE {"
29                          + "  GRAPH <%s> REVISION \"%s\" { :Adam :knows ?a. }%n"
30                          + "  GRAPH <%s> REVISION \"%s\" {?a :address ?address. }%n"
31                          + "} %n"
32                          + "ORDER BY ?address",
33                  ds1.graphName, ds1.revisions.get(rev1), ds2.graphName, ds2.revisions.get(rev2));
34      }
35  
36      @BeforeClass
37      public static void setUpBeforeClass() throws Exception {
38          Config.readConfig("r43ples.test.conf");
39          ds1 = SampleDataSet.createSampleDataset1();
40          ds2 = SampleDataSet.createSampleDataset2();
41      }
42  
43      /**
44       * @throws IOException
45       * @throws InternalErrorException
46       */
47      @Test
48      public final void testMultipleGraphs() throws IOException, InternalErrorException {
49          Endpoint ep = new Endpoint();
50          String result = ep.sparql("text/turtle", get_query_template("master-1", "master-1")).getEntity().toString();
51          String expected = ResourceManagement.getContentFromResource("response-TwoGraphs-1-1.ttl");
52          assertTrue(check_isomorphism(result, expected));
53  
54          result = ep.sparql("text/turtle", get_query_template("master-2", "master-1")).getEntity().toString();
55          expected = ResourceManagement.getContentFromResource("response-TwoGraphs-2-1.ttl");
56          assertTrue(check_isomorphism(result, expected));
57  
58          result = ep.sparql("text/turtle", get_query_template("master-2", "master-2")).getEntity().toString();
59          expected = ResourceManagement.getContentFromResource("response-TwoGraphs-2-2.ttl");
60          assertTrue(check_isomorphism(result, expected));
61      }
62  
63      /**
64       * @throws IOException
65       * @throws InternalErrorException
66       */
67      @Test
68      public final void testMultipleGraphsQueryRewriting() throws IOException, InternalErrorException {
69          String result = ep.sparql("text/turtle", get_query_template("master-1", "master-1"), true).getEntity().toString();
70          String expected = ResourceManagement.getContentFromResource("response-TwoGraphs-1-1.ttl");
71          assertTrue(check_isomorphism(result, expected));
72  
73          result = ep.sparql("text/turtle", get_query_template("master-2", "master-1"), true).getEntity().toString();
74          expected = ResourceManagement.getContentFromResource("response-TwoGraphs-2-1.ttl");
75          assertTrue(check_isomorphism(result, expected));
76  
77          result = ep.sparql("text/turtle", get_query_template("master-2", "master-2"), true).getEntity().toString();
78          expected = ResourceManagement.getContentFromResource("response-TwoGraphs-2-2.ttl");
79          assertTrue(check_isomorphism(result, expected));
80      }
