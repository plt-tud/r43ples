43      /**
44       * Checks the isomorphism between two sets of triples with given specific serialisations.
45       *
46       * @param result_set          the result set
47       * @param result_set_format   the result set format
48       * @param expected_set        the expected set
49       * @param expected_set_format the expected set format
50  	 * @return true if the two models are isomorphic
51  	 */
52  	public boolean check_isomorphism(String result_set, String result_set_format, String expected_set, String expected_set_format) {
53  		return check_isomorphism(JenaModelManagement.readStringToJenaModel(result_set, result_set_format), JenaModelManagement.readStringToJenaModel(expected_set, expected_set_format));
54  	}

57  	/**
58  	 * Checks the isomorphism between two models.
59  	 * 
60  	 * @param result_model the result model
61  	 * @param expected_model the expected model
62  	 * @return true if the two models are isomorphic
63  	 */
64  	public boolean check_isomorphism(Model result_model, Model expected_model) {
65  		// Check isomorphism
66  		if (result_model.isIsomorphicWith(expected_model)) {
67  			return true;
68  		} else {
69  			// Get the differences between the models
70  
71  			// Resulting model contains all statements which are in the result_model but not in the expected_model		
72              logger.error("The following statements are in the result_model but not in the expected_model: \n"
73                      + JenaModelManagement.convertJenaModelToNTriple(result_model.difference(expected_model)));
74  			// Resulting model contains all statements which are in the expected_model but not in the result_model	
75              logger.error("The following statements are in the expected_model but not in the result_model: \n"
76                      + JenaModelManagement.convertJenaModelToNTriple(expected_model.difference(result_model)));
77  
78              return false;
79  		}
80  	}

82      /**
83       * Remove all statements with property prov:atTime from given model
84       *
85       * @param model model which should remove all prov:atTime statements
86       * @return model with removed statements
87       */
88      public Model removeTimeStampFromModel(Model model) {
89          // Remove timestamp for test
90          Property provAtTime = model.getProperty("http://eatld.et.tu-dresden.de/rmo#timeStamp");
91          StmtIterator stmtIterator = model.listStatements(null, provAtTime, (RDFNode) null);
92          model.remove(stmtIterator);
93  
94          return model;
95      }

97  	/**
98  	 * Checks if the result and the expected graphs are isomorphic by using assertion.
99  	 *
100 	 * @param result the result graph as turtle
101 	 * @param expected the expected graph as turtle
102 	 */
103     public void assertIsomorphism(String result, String expected) {
104 		Model model_result = JenaModelManagement.readTurtleStringToJenaModel(result);
105 		Model model_expected = JenaModelManagement.readTurtleStringToJenaModel(expected);
106 
107 		this.removeTimeStampFromModel(model_result);
108 		this.removeTimeStampFromModel(model_expected);
109 
110 		Assert.assertTrue(check_isomorphism(model_result, model_expected));
111 	}

123 	/**
124 	 * Checks the existence of all necessary graphs by using assertion.
125 	 *
126 	 * @param list the list of graph URIs
127 	 */
128 	public void assertListAll(LinkedList<String> list, TripleStoreInterface tripleStoreInterface) {
129 		Iterator<String> iterator = tripleStoreInterface.getGraphs();
130 		while (iterator.hasNext()) {
131 			String next = iterator.next();
132 			if (list.contains(next)) {
133 				list.remove(next);
134 			} else {
135 				Assert.fail("The graph URI <" + next + "> is missing in the list of expected URIs.");
136 			}
137 		}
138 		if (!list.isEmpty()) {
139 			Assert.fail("The following URIs can not be found in triple store: " + list.toArray().toString());
140 		}
141 	}
