34      public final void testPathToLeaf() throws InternalErrorException {
35  
36          RevisionGraph revisionGraph = new RevisionGraph(ds1.graphName);
37          Revision revision = revisionGraph.getRevision("2");
38          SimplePathCalculation pathCalc = new SimplePathCalculation(revisionGraph);
39          Path test = pathCalc.getPathToRevisionWithFullGraph(revision);
40  
41          Assert.assertEquals(4, test.getRevisionPath().size());
42          Assert.assertEquals("2", test.getRevisionPath().get(0).getRevisionIdentifier());
43          Assert.assertEquals("3", test.getRevisionPath().get(1).getRevisionIdentifier());
44          Assert.assertEquals("4", test.getRevisionPath().get(2).getRevisionIdentifier());
45          Assert.assertEquals("5", test.getRevisionPath().get(3).getRevisionIdentifier());
46  
47          
48          RevisionGraph revisionGraph2 = new RevisionGraph(ds2.graphName);
49          Revision revision2 = revisionGraph2.getRevision("1");
50          SimplePathCalculation pathCalc2 = new SimplePathCalculation(revisionGraph2);
51          Path test2 = pathCalc2.getPathToRevisionWithFullGraph(revision2);
52          Assert.assertEquals(3, test2.getRevisionPath().size());
53          Assert.assertEquals("1", test2.getRevisionPath().get(0).getRevisionIdentifier());
54          Assert.assertEquals("2", test2.getRevisionPath().get(1).getRevisionIdentifier());
55          Assert.assertEquals("5", test2.getRevisionPath().get(2).getRevisionIdentifier());
56      }
